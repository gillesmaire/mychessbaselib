34template <class TElem, size_t UNDOMAX> class UndoRedo {
35 std::vector<TElem*> undo_;
36 std::vector<TElem*> redo_;
37
38public:
39 ~UndoRedo() { clear(); }
40 void clear() {
41 clear(undo_);
42 clear(redo_);
43 }
44 size_t undoSize() const { return undo_.size(); }
45 size_t redoSize() const { return redo_.size(); }
46
52 void store(TElem* current) {
53 clear(redo_);
54 undo_.push_back(current->clone());
55 if (undo_.size() > UNDOMAX) {
56 delete undo_.front();
57 undo_.erase(undo_.begin());
58 }
59 }
60
68 TElem* undo(TElem* current) { return doUndoRedo(undo_, redo_, current); }
69
77 TElem* redo(TElem* current) { return doUndoRedo(redo_, undo_, current); }
78
79private:
80 // Store current into cont2; remove and return the last element of cont1
81 template <typename TCont>
82 TElem* doUndoRedo(TCont& cont1, TCont& cont2, TElem* current) {
83 if (cont1.empty())
84 return current;
85
86 if (cont2.empty() || cont2.back() != current)
87 cont2.push_back(current);
88
89 auto res = cont1.back();
90 cont1.pop_back();
91 return res;
92 }
93
94 template <typename TCont> void clear(TCont& cont) {
95 for (auto& e : cont)
96 delete e;
97 cont.clear();
98 }
99};

109template <class T, size_t CHUNKSHIFT> class VectorChunked {
110 std::vector<T*> chunks_;
111 size_t size_ = 0;
112
113 static constexpr size_t low_mask = ((1ULL << CHUNKSHIFT) - 1);
114
115public:
116 VectorChunked() = default;
117 VectorChunked(const VectorChunked&) = delete;
118 VectorChunked& operator=(const VectorChunked&) = delete;
119 ~VectorChunked() {
120 for (auto& chunk : chunks_)
121 delete[] chunk;
122 }
123
124 const T& operator[](size_t idx) const {
125 assert(idx < size_);
126 return chunks_[idx >> CHUNKSHIFT][idx & low_mask];
127 }
128 T& operator[](size_t idx) {
129 assert(idx < size_);
130 return chunks_[idx >> CHUNKSHIFT][idx & low_mask];
131 }
132
133 size_t capacity() const { return chunks_.size() << CHUNKSHIFT; }
134
139 size_t contiguous(size_t pos) const {
140 assert(pos < size());
141 return 1 + (~pos & low_mask);
142 }
143
146 size_t next_contiguous(size_t nElements) const {
147 const auto offset = size();
148 const auto capacity = this->capacity();
149 return capacity - offset < nElements // Doesn't fit in the current chunk
150 ? capacity
151 : offset;
152 }
153
159 void append(const T* src, size_t srcSize, size_t offset) {
160 assert(offset >= size());
161 resize(offset + srcSize);
162 assert(contiguous(offset) >= srcSize);
163 std::copy_n(src, srcSize, &operator[](offset));
164 }
165
170 const T* append(const T* src, size_t srcSize) {
171 const auto offset = next_contiguous(srcSize);
172 append(src, srcSize, offset);
173 return &operator[](offset);
174 }
175
176 void push_back(const T& e) {
177 size_t idx = size_;
178 resize(size_ + 1);
179 operator[](idx) = e;
180 }
181
182 void resize(size_t count) {
183 size_ = count;
184 size_t newSize = (count > 0) ? 1 + (count >> CHUNKSHIFT) : 0;
185 size_t chunksSz = chunks_.size();
186 if (newSize == chunksSz)
187 return;
188
189 if (newSize > chunksSz) {
190 chunks_.resize(newSize);
191 for (auto i = chunksSz; i < newSize; ++i) {
192 chunks_[i] = new T[1ULL << CHUNKSHIFT];
193 }
194 } else {
195 for (auto i = newSize; i < chunksSz; ++i) {
196 delete[] chunks_[i];
197 }
198 chunks_.resize(newSize);
199 }
200 }
201
202 size_t size() const { return size_; }
203};
