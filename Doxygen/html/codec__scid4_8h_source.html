91 ByteBuffer getGameData(uint64_t offset, uint32_t length) final {
92 if (offset >= gfile_.size())
93 return {nullptr, 0};
94 if (length >= LIMIT_GAMELEN)
95 return {nullptr, 0};

97 if (gfile_.pubseekpos(offset) == -1)
98 return {nullptr, 0};
99 if (gfile_.sgetn(gamecache_, length) != std::streamsize(length))
100 return {nullptr, 0};

155 std::pair<errorT, uint64_t> dyn_addGameData(const byte* src,
156 size_t length) {
157 ASSERT(src != 0);
158 const char* data = reinterpret_cast<const char*>(src);
159
160 if (length >= LIMIT_GAMELEN)
161 return std::make_pair(ERROR_GameLengthLimit, 0);
162
163 // The SCID4 format uses 32-bits to store games' offset.
164 uint64_t offset = gfile_.size();
165 if (offset >= LIMIT_GAMEOFFSET - length)
166 return std::make_pair(ERROR_OffsetLimit, 0);
167
168 // The SCID4 format stores games into blocks of 128KB.
169 // If the current block does not have enough space, we fill it with
170 // random data and use the next one.
171 uint64_t blockSpace = LIMIT_GAMELEN - (offset % LIMIT_GAMELEN);
172 if (blockSpace < length) {
173 errorT err = gfile_.append(data, blockSpace);
174 if (err != OK)
175 return std::make_pair(err, 0);
176 offset += blockSpace;
177 }
178
179 errorT err = gfile_.append(data, length);
180 return std::make_pair(err, offset);
181 }
182
192 std::pair<errorT, idNumberT> dyn_addName(nameT nt, const char* name) {
193 if (std::string_view(name).size() > LIMIT_NAMELEN)
194 return {ERROR_NameTooLong, 0};
195
196 static constexpr auto limit_unique_names = [] {
197 std::array<unsigned long long, NUM_NAME_TYPES> res;
198 res[NAME_PLAYER] = 1048575; // Maximum of 2^20 -1 = 1,048,575
199 res[NAME_EVENT] = 524287; // Maximum of 2^19 -1 = 524,287
200 res[NAME_SITE] = 524287; // Maximum of 2^19 -1 = 524,287
201 res[NAME_ROUND] = 262143; // Maximum of 2^18 -1 = 262,143
202 return res;
203 }();
204 if (nb_->namebase_size(nt) < limit_unique_names[nt])
205 return {OK, nb_->namebase_find_or_add(nt, name)};
206
207 idNumberT id;
208 if (nb_->FindExactName(nt, name, &id) == OK)
209 return {OK, id};
210
211 return {ERROR_NameLimit, 0};
212 }

242 errorT addGameNamesAndData(IndexEntry& ie, TagRoster const& tags,
243 const byte* srcData, size_t dataLen) {
244 if (!ie.isChessStd())
245 return ERROR_CodecChess960;
246
247 auto errNames = tags.map(
248 ie, [&](auto nt, auto name) { return dyn_addName(nt, name); });
249 if (errNames)
250 return errNames;
251
252 auto [err, offset] = dyn_addGameData(srcData, dataLen);
253 if (!err) {
254 ie.SetOffset(offset);
255 ie.SetLength(dataLen);
256 }
257 return err;
258 }
