137 std::pair<errorT, uint64_t> dyn_addGameData(const byte* src,
138 size_t length) {
139 ASSERT(src != 0);
140
141 if (length >= LIMIT_GAMELEN)
142 return std::make_pair(ERROR_GameLengthLimit, 0);
143
144 const auto offset = v_.next_contiguous(length);
145 if (offset >= LIMIT_GAMEOFFSET)
146 return std::make_pair(ERROR_OffsetLimit, 0);
147
148 v_.append(src, length, offset);
149 return {OK, offset};
150 }

161 std::pair<errorT, idNumberT> dyn_addName(nameT nt, const char* name) {
162 static constexpr auto limit_unique_names = [] {
163 std::array<unsigned long long, NUM_NAME_TYPES> res;
164 res[NAME_PLAYER] = LIMIT_UNIQUENAMES_PLAYER_EVENT;
165 res[NAME_EVENT] = LIMIT_UNIQUENAMES_PLAYER_EVENT;
166 res[NAME_SITE] = LIMIT_UNIQUENAMES_SITE;
167 res[NAME_ROUND] = LIMIT_UNIQUENAMES_ROUND;
168 return res;
169 }();
170 if (nb_->namebase_size(nt) < limit_unique_names[nt])
171 return {OK, nb_->namebase_find_or_add(nt, name)};
172
173 idNumberT id;
174 if (nb_->FindExactName(nt, name, &id) == OK)
175 return {OK, id};
176
177 return {ERROR_NameLimit, 0};
178 }

212private:
215 errorT addGameNamesAndData(IndexEntry& ie, TagRoster const& tags,
216 const byte* srcData, size_t dataLen) {
217 auto errNames = tags.map(
218 ie, [&](auto nt, auto name) { return dyn_addName(nt, name); });
219 if (errNames)
220 return errNames;
221
222 auto [err, offset] = dyn_addGameData(srcData, dataLen);
223 if (!err) {
224 ie.SetOffset(offset);
225 ie.SetLength(dataLen);
226 }
227 return err;
228 }
