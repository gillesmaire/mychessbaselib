349 }

413 // Add a tag.
414 // For the tags that cannot be duplicated (like Event or White), the
415 // previous value will be overwritten.
416 std::string& addTag(std::string_view tag, std::string_view value) {
417 if (auto overwrite = find_std_tag(tag))
418 return overwrite->assign(value);
419
420 return extraTags_.emplace_back(tag, value).second;
421 }
422
423 // Change the value of a tag (add the tag if it wasn't present).
424 template <typename... Args>
425 std::string& assignTagValue(std::string_view tag, Args&&... args) {
426 auto value = find_std_tag(tag);
427 if (!value) {
428 auto it = std::find_if(
429 extraTags_.begin(), extraTags_.end(),
430 [&](auto const& elem) { return elem.first == tag; });
431 if (it != extraTags_.end()) {
432 value = &it->second;
433 } else {
434 value = &extraTags_.emplace_back(tag, std::string()).second;
435 }
436 }
437 return value->assign(std::forward<Args>(args)...);
438 }

530 char strBuf[256];
531 visitor("Event", GetEventStr());
532 visitor("Site", GetSiteStr());
533 date_DecodeToString(GetDate(), strBuf);
534 visitor("Date", strBuf);
535 visitor("Round", GetRoundStr());
536 visitor("White", GetWhiteStr());
537 visitor("Black", GetBlackStr());
538 visitor("Result", RESULT_LONGSTR[GetResult()]);
539

541 std::string rType = "White";

552 visitor("ECO", strBuf);

566 visitor("FEN", strBuf);

599template <typename TCont>
600inline void collectPositions(Game& game, TCont& dest) {
601 do {
602 if (game.AtVarStart() && !game.AtStart())
603 continue;
604
605 dest.emplace_back();
606 auto& gamepos = dest.back();
607 gamepos.RAVdepth = game.GetVarLevel();
608 gamepos.RAVnum = game.GetVarNumber();
609 char strBuf[256];
610 game.currentPos()->PrintFEN(strBuf);
611 gamepos.FEN = strBuf;
612 for (byte* nag = game.GetNags(); *nag; nag++) {
613 gamepos.NAGs.push_back(*nag);
614 }
615 gamepos.comment = game.GetMoveComment();
616 game.GetPrevSAN(strBuf);
617 gamepos.lastMoveSAN = strBuf;
618
619 } while (game.MoveForwardInPGN() == OK);
620}

628inline std::vector<GamePos> collectPositions(Game& game) {
629 std::vector<GamePos> res;
630 game.MoveToStart();
631 collectPositions(game, res);
632 return res;
633}
