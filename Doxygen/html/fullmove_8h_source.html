25class FullMove {
26 // ** Lower 16 bits are compatible with Stockfish's Move **
27 // bits 0- 5: destination square (from 0 to 63)
28 // bits 6-11: origin square (from 0 to 63)
29 // bits 12-13: promotion piece type -2 (from QUEEN-2 to KNIGHT-2)
30 // bits 14-15: special move flag: promotion (1), en passant (2), castling (3)
31
32 // ** Info for undoing the move **
33 // bits 16-17: castling flags - TODO
34 // bits 18-20: enpassant file - TODO
35 // bits 21-23: captured pieceT
36
37 // ** Info for direct SAN conversion **
38 // bits 24-26: moving pieceT
39 // bit 27: black to move
40 // bit 28: ambiguous move, insert from fyle
41 // bit 29: ambiguous move, insert from rank
42 // bit 30: check
43
44 // ** TODO: Use this flag to embed tags, variations, etc.. in a move stream
45 // bit 31: special flag
46 uint32_t m_;
47
48public:
49 constexpr FullMove(uint32_t m = 0) : m_(m){};
50
51 FullMove(colorT c, squareT kingSq, squareT rookSq)
52 // Castle: encoding as king to rook allows the undoing of chess360 moves
53 : FullMove(c, kingSq, rookSq, KING) {
54 m_ |= (3 << 14);
55 }
56
57 FullMove(colorT c, squareT from, squareT to, pieceT pt) {
58 m_ = to | (from << 6) | (pt << 24) | (c << 27);
59 }
60
61 bool operator==(FullMove const& f) const { return m_ == f.m_; }
62
63 // Special moves:
64 // NONE: encoded as 0 (from 0 to 0);
65 // NULL: encoded as 65 (from 1 to 1);
66 // PROMO: encoded setting the special move flag to 1
67 // ENPASSANT: encoded setting the special move flag to 2
68 // CASTLING: encoded setting the special move flag to 3, from is the square
69 // of the king and to is the square of the rook.
70 // If from < to it is castling king side.
71 explicit operator bool() const { return m_ != 0; }
72 bool isNull() const { return m_ == 0b01000001; }
73 bool isPromo() const { return (m_ & (3 << 14)) == (1 << 14); }
74 bool isEnpassant() const { return (m_ & (3 << 14)) == (2 << 14); }
75 bool isCastle() const { return (m_ & (3 << 14)) == (3 << 14); }
76
77 squareT getTo() const { return m_ & 0x3F; }
78 squareT getFrom() const { return (m_ >> 6) & 0x3F; }
79 pieceT getPiece() const { return (m_ >> 24) & 0x07; }
80 colorT getColor() const { return (m_ >> 27 & 1) ? BLACK : WHITE; }
81 pieceT getPromo() const { return ((m_ >> 12) & 0x03) +2; }
82 pieceT getCaptured() const { return (m_ >> 21) & 0x07; }
83 std::string getSAN() const {
84 std::string res;
85 const auto to = getTo();
86 const auto from = getFrom();
87 if (to == 0 && from == 0) return "--";
88 if (isCastle()) {
89 res = (to > from) ? "O-O" : "O-O-O";
90 bool check = (m_ >> 30) & 1;
91 if (check)
92 res += "+";
93 return res;
94 }
95 bool fromFyle = (m_ >> 28) & 1;
96 bool fromRank = (m_ >> 29) & 1;
97 bool check = (m_ >> 30) & 1;
98 bool capture = (getCaptured() != 0);
99
100 switch (getPiece()) {
101 case BISHOP: res += "B"; break;
102 case KNIGHT: res += "N"; break;
103 case ROOK: res += "R"; break;
104 case QUEEN: res += "Q"; break;
105 case KING: res += "K"; break;
106 default: //PAWN
107 if (capture) res += 'a' + (from % 8);
108 }
109 if (fromFyle) res += 'a' + (from % 8);
110 if (fromRank) res += '1' + (from / 8);
111 if (capture) res += "x";
112 res += 'a' + (to % 8);
113 res += '1' + (to / 8);
114 if (isPromo()) {
115 switch (getPromo()) {
116 case BISHOP: res += "=B"; break;
117 case KNIGHT: res += "=N"; break;
118 case ROOK: res += "=R"; break;
119 case QUEEN: res += "=Q"; break;
120 }
121 }
122 if (check) res += "+";
123 return res;
124 }
125
126 void setPromo(pieceT promo) {
127 ASSERT(promo == QUEEN || promo == ROOK || promo == BISHOP ||
128 promo == KNIGHT);
129 m_ |= ((promo - 2) << 12) | (1 << 14);
130 }
131 void setCapture(pieceT piece, bool enPassant) {
132 m_ |= ((piece & 0x07) << 21);
133 if (enPassant) m_ |= (2 << 14);
134 }
135 void setAmbiguity(bool fyle, bool rank) {
136 m_ &= ~(3 << 28);
137 if (fyle)
138 m_ |= (1 << 28);
139 if (rank)
140 m_ |= (1 << 29);
141 }
142 void setCheck() { m_ |= (1 << 30); }
143};
