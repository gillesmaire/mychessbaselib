34class Index
35{
36private:
37 // The complete index will be loaded in memory and can be pretty huge.
38 // To avoid the slow reallocation when adding games we split the data in chunks.
39 // CHUNKSHIFT is the base-2 logarithm of the number of index entries allocated as one chunk.
40 // i.e 16 = 2^16 = 65536 (total size of one chunk: 65536*48 = 3MB)
41 VectorChunked<IndexEntry, 16> entries_; // A two-level array of the entire index.
42 int nInvalidNameId_;
43
44 friend class CodecSCID4;
45 friend class CodecSCID5;
46
47public:
48 Index() { Init(); }
49
50 void Close() { Init(); }
51
52 const IndexEntry* GetEntry (gamenumT g) const {
53 ASSERT(g < GetNumGames());
54 return &(entries_[g]);
55 }
56
66 int GetBadNameIdCount() const { return nInvalidNameId_; }
67
71 gamenumT GetNumGames() const {
72 return static_cast<gamenumT>(entries_.size());
73 }
74
75 void addEntry(const IndexEntry& ie) {
76 entries_.push_back(ie);
77 }
78
79 void replaceEntry(const IndexEntry& ie, gamenumT replaced) {
80 ASSERT(replaced < this->GetNumGames());
81
82 entries_[replaced] = ie;
83 }
84
85private:
86 void Init() {
87 nInvalidNameId_ = 0;
88 entries_.resize(0);
89 }
90};
