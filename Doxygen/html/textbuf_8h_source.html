93inline void
94TextBuffer::AddChar (char ch)
95{
96 if (HasTranslations && !PausedTranslations) {
97 byte b = (byte) ch;
98 const char * str = Translation[b];
99 if (str != NULL) {
100 const char * s = str;
101 while (*s) {
102 *Current++ = *s++;
103 ByteCount++;
104 }
105 return;
106 }
107 }
108 *Current = ch;
109 Current++;
110 ByteCount++;
111}
