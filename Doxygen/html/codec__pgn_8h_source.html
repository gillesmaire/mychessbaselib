87 errorT parseNext(Game& game) {
88 const auto verge = 3 * (nRead_ / 4);
89 if (nParsed_ > verge && nRead_ == buf_.size()) {
90 nParsed_ -= verge;
91 nRead_ -= verge;
92 std::copy_n(buf_.data() + verge, nRead_, buf_.data());
93 nRead_ += file_.sgetn(buf_.data() + nRead_, verge);
94 }
95
96 game.Clear();
97 PgnVisitor visitor(game);
98 auto parse = pgn::parse_game(
99 {buf_.data() + nParsed_, buf_.data() + nRead_}, visitor);
100
101 bool eof = (nRead_ - nParsed_ == parse.first);
102 if (eof && nRead_ == buf_.size()) {
103 // Reached the end of input, but the file contains more bytes.
104 if (nRead_ <= 128 * 1024 * 1024) {
105 // Double the buffer size and retry.
106 buf_.resize(nRead_ * 2);
107 nRead_ += file_.sgetn(buf_.data() + nRead_, nRead_);
108 return parseNext(game);
109 }
110 // Abort
111 nRead_ = nParsed_ = 0;
112 parseLog_.log.append("PGN parsing aborted.\n");
113 return ERROR_NotFound;
114 }
115
116 nParsed_ += parse.first;
117 parseLog_.logGame(parse.first, visitor);
118 if (eof && !parse.second && *game.GetMoveComment() == '\0')
119 return ERROR_NotFound;
120
121 return OK;
122 }
