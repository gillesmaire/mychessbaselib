49inline constexpr size_t MAX_TAG_LEN = 240;
50inline constexpr std::string_view commonTags[] = {
51 // 241, 242: Country
52 "WhiteCountry", "BlackCountry",
53 // 243: Annotator
54 "Annotator",
55 // 244: PlyCount
56 "PlyCount",
57 // 245: EventDate (plain text encoding)
58 "EventDate",
59 // 246, 247: Opening, Variation
60 "Opening", "Variation",
61 // 248-250: Setup and Source
62 "Setup", "Source", "SetUp"
63 // 252-254: spare for future use
64 // 255: Reserved for compact EventDate encoding
65};

67template <typename SourceT, typename DestT>
68void encodeTags(const SourceT& tagList, DestT& dest) {
69 for (auto& tag : tagList) {
70 if (tag.first.length() == 0)
71 continue; // Cannot store empty tag names
72
73 const auto it = std::find(commonTags, std::end(commonTags), tag.first);
74 if (it != std::end(commonTags)) {
75 // Common tags are stored with just their 1-byte value [241:250]
76 const auto tagnum = std::distance(commonTags, it) + MAX_TAG_LEN + 1;
77 dest.emplace_back(static_cast<unsigned char>(tagnum));
78 } else {
79 // Other tags are stored as 1-byte length [1:240] + the tag name.
80 const auto tag_name = tag.first.data();
81 const auto length = std::min(tag.first.length(), MAX_TAG_LEN);
82 dest.emplace_back(static_cast<unsigned char>(length));
83 dest.insert(dest.end(), tag_name, tag_name + length);
84 }
85
86 // The value is stored as 1-byte length [0:255] + the data.
87 const auto value = tag.second.data();
88 const auto valueLen = std::min<size_t>(tag.second.length(), 255);
89 dest.emplace_back(static_cast<unsigned char>(valueLen));
90 dest.insert(dest.end(), value, value + valueLen);
91 }
92 dest.emplace_back(0);
93}

103template <typename DestT>
104void encodeStartBoard(bool promoFlag, bool underpromoFlag, const char* FEN,
105 DestT& dest) {
106 char flags = 0;
107 if (FEN) {
108 flags += 1;
109 }
110 if (promoFlag) {
111 flags += 2;
112 }
113 if (underpromoFlag) {
114 flags += 4;
115 }
116 dest.emplace_back(flags);
117 if (FEN) {
118 const auto len = std::strlen(FEN) + 1; // Include the null char
119 dest.insert(dest.end(), FEN, FEN + len);
120 }

142 template <typename FuncT> errorT decodeTags(FuncT fn) {
143 if (data_ == end_)
144 return ERROR_BufferRead;
145
146 auto it = data_;
147 for (;;) {
148 const auto tagLen = *it++;
149 if (tagLen == 0) {
150 data_ = it;
151 return OK; // Reached the end of the tags section
152 }
153
154 const char* tag = nullptr;
155 size_t tagID = 0;
156 if (tagLen > MAX_TAG_LEN) {
157 // A common tag name, not explicitly stored
158 tagID = tagLen - MAX_TAG_LEN - 1;
159 } else {
160 tag = reinterpret_cast<const char*>(it);
161 it += tagLen;
162 }
163
164 if (it >= end_)
165 return ERROR_Decode;
166
167 // 255 was a special 3-bytes encoding of EventDate used in SCID2
168 const auto valueLen = (tagLen != 255) ? *it++ : 3;
169 const char* value = reinterpret_cast<const char*>(it);
170 it += valueLen;
171 if (it >= end_)
172 return ERROR_Decode;
173
174 if (tag) {
175 fn(std::string_view(tag, tagLen),
176 std::string_view(value, valueLen));
177 } else if (tagID < 10) {
178 fn(commonTags[tagID], std::string_view(value, valueLen));
179 }
180 }
181 }

192 std::pair<errorT, const char*> decodeStartBoard() {
193 if (data_ == end_)
194 return {ERROR_Decode, nullptr};
195
196 const auto flags = *data_++;
197 if ((flags & 1) == 0)
198 return {OK, nullptr};
199
200 if (const auto FEN = GetTerminatedString())

203 return {ERROR_Decode, nullptr};

207 const char* GetTerminatedString() {
208 const char* res = reinterpret_cast<const char*>(data_);
209 data_ = std::find(data_, end_, 0);
210 if (data_ == end_)
211 return nullptr;
212
213 ++data_; // skip the null char
214 return res;

221 template <typename MoveFn, typename CommentFn, typename VariationFn,
222 typename NagFn>
223 std::pair<errorT, unsigned char>
224 nextMove(int varDepth, MoveFn acceptMove, CommentFn commentMarker,
225 VariationFn changeVar, NagFn addNag) {
226 // The king has always index 0 and only 11 possible moves (8 destination
227 // squares, 2 castle moves and the null move). The unused codes 11-15
228 // represent special markers for nags, comments and variations.
229 // Note: 2-bytes queen moves use the second byte as (64 + destination
230 // square) to avoid interferences with these markers. However the byte
231 // immediatly after the ENCODE_NAG can have any possible value.
232 enum {
233 ENCODE_NAG = 11,
234 ENCODE_COMMENT,
235 ENCODE_START_MARKER,
236 ENCODE_END_MARKER,
237 ENCODE_END_GAME
238 };
239
240 auto it = data_;
241 for (; it != end_; ++it) {
242 switch (*it) {
243 case ENCODE_NAG:
244 if (++it == end_) {
245 return {ERROR_Decode, 0}; // ERROR: missing nag
246 }
247 if (!addNag(*it)) {
248 return {ERROR_Decode, 0}; // ERROR: marker decoding
249 }
250 continue;
251
252 case ENCODE_COMMENT:
253 commentMarker();
254 continue;
255
256 case ENCODE_START_MARKER:
257 ++varDepth;
258 if (!changeVar(true)) {
259 return {ERROR_Decode, 0}; // ERROR: variation
260 }
261 continue;
262
263 case ENCODE_END_MARKER:
264 if (varDepth == 0) {
265 return {ERROR_Decode, 0}; // ERROR: end marker in main line
266 }
267 --varDepth;
268 if (!changeVar(false)) {
269 return {ERROR_Decode, 0}; // ERROR: variation
270 }
271 continue;
272
273 case ENCODE_END_GAME:
274 if (varDepth != 0) {
275 return {ERROR_Decode, 0}; // ERROR: unexpected end of game
276 }
277 data_ = ++it;
278 return {ERROR_EndOfMoveList, 0}; // SUCCESS: end of game
279
280 default:
281 if (acceptMove(varDepth)) {
282 data_ = it;
283 return {OK, *data_++}; // SUCCESS
284 }
285 }
286 }
287 return {ERROR_Decode, 0}; // ERROR: missing ENCODE_END_GAME
288 }

315 std::pair<int, pieceT> decodeMove(colorT toMove, pieceT movingPiece,
316 squareT from, unsigned char moveCode) {
317 moveCode &= 0x0F;
318 switch (movingPiece) {
319 case PAWN: {
320 static const pieceT promoPiece[] = {
321 INVALID_PIECE, INVALID_PIECE, INVALID_PIECE, QUEEN,
322 QUEEN, QUEEN, ROOK, ROOK,
323 ROOK, BISHOP, BISHOP, BISHOP,
324 KNIGHT, KNIGHT, KNIGHT, INVALID_PIECE};
325 static const int8_t sqdiff[] = {7, 8, 9, 7, 8, 9, 7, 8,
326 9, 7, 8, 9, 7, 8, 9, 16};
327 int to = (toMove == WHITE) ? from + sqdiff[moveCode]
328 : from - sqdiff[moveCode];
329 return {to, promoPiece[moveCode]};
330 }
331 case BISHOP: {
332 int fylediff = square_Fyle(moveCode) - square_Fyle(from);
333 int to = (moveCode >= 8) ? from - 7 * fylediff
334 : from + 9 * fylediff;
335 return {to, INVALID_PIECE};
336 }
337 case KNIGHT: {
338 static const int8_t sqdiff[] = {0, -17, -15, -10, -6, 6, 10, 15,
339 17, 0, 0, 0, 0, 0, 0, 0};
340 return {from + sqdiff[moveCode], INVALID_PIECE};
341 }
342 case QUEEN:
343 if (moveCode == square_Fyle(from)) { // 2 BYTES MOVE
344 int to = (data_ != end_) ? *data_++ : 0;
345 return {to - 64, INVALID_PIECE};
346 }
347 [[fallthrough]];
348 case ROOK: {
349 int to = (moveCode >= 8) // a vertical move
350 ? square_Make(square_Fyle(from), (moveCode - 8))
351 : square_Make(moveCode, square_Rank(from));
352 return {to, INVALID_PIECE};
353 }
354 case KING:
355 if (moveCode == 0) // NULL MOVE
356 return {from, PAWN};
357
358 if (moveCode <= 8) {
359 static const int8_t sqdiff[] = {0, -9, -8, -7, -1, 1, 7, 8, 9};
360 return {from + sqdiff[moveCode], INVALID_PIECE};
361 }
362
363 if (moveCode == 9) // CASTLE QUEENSIDE
364 return {from, QUEEN};
365
366 if (moveCode == 10) // CASTLE KINGSIDE
367 return {from, KING};
368 }
369
370 return {from, INVALID_PIECE}; // decode error
371 }
