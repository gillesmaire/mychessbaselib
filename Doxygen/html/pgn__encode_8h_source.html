35// We want to split the PGN text in lines to make it more readable, but we do
36// not want to insert extra newline chars inside comments or tag values.
37// This implies that even very long comment would stay on a single line if the
38// user didn't insert newline chars himself.
39// However it is possible to set @e hard_len (i.e to 1024) to allow converting
40// spaces to newline chars in lines longer than @e hard_len.
41template <int desired_len = 80, char breakpoint_char = '\0', int hard_len = 0,
42 typename Iter>
43Iter break_lines(Iter begin, Iter end) {
44 auto line_first_char = begin;
45 auto last_breakpoint = begin;
46 auto it = begin;
47 while (true) {
48 it = find_if(it, end, [&](char ch) {
49 return ch == '\n' || ch == breakpoint_char;
50 });
51
52 // Change the last breakpoint to newline char if the line would exceed
53 // the desired length and there weren't newline chars (for example in
54 // comments) beetween this and the last breakpoint.
55 if (std::distance(line_first_char, it) > desired_len &&
56 last_breakpoint > line_first_char) {
57 *last_breakpoint = '\n';
58 line_first_char = last_breakpoint + 1;
59 }
60
61 // If a secondary line length was requested, try to convert spaces to
62 // newline chars (this is not desiderable, but old software may use
63 // limited fixed size buffer when reading PGNs).
64 if (hard_len != 0 && std::distance(line_first_char, it) > hard_len) {
65 line_first_char = break_lines<hard_len, ' '>(line_first_char, it);
66 }
67
68 if (it == end)
69 break;
70
71 if (*it == '\n') {
72 line_first_char = ++it;
73 } else /* *it == breakpoint_char */ {
74 last_breakpoint = it;
75 *it++ = ' ';
76 }
77 }
78 return line_first_char;
79}

81// Escape quote and backslash chars according to the PGN standard:
82// "A quote inside a string is represented by the backslash immediately followed
83// by a quote. A backslash inside a string is represented by two adjacent
84// backslashes."
85// @param str: the string containing the chars to be escaped.
86// @param pos: start of the substring of @e str to be processed.
87template <typename TCont>
88void escape_string(TCont& str, typename TCont::size_type pos) {
89 auto it = str.begin() + pos;
90 while (true) {
91 it = std::find_if(it, str.end(),
92 [](char ch) { return ch == '\\' || ch == '\"'; });
93 if (it != str.end())
94 it = str.insert(it, '\\') + 2;
95 else
96 break;
97 }
98}

100// Encode a tag pair according to the PGN standard.
101// "A tag pair is composed of four consecutive tokens: a left bracket token, a
102// symbol token, a string token, and a right bracket token. The symbol token is
103// the tag name and the string token is the tag value associated with the tag
104// name. There are no white space characters between the left bracket and the
105// tag name, there are no white space characters between the tag value and the
106// right bracket, and there is a single space character between the tag name and
107// the tag value."
108// @param unknown_to_question_mark: if true, and a Seven Tag Roster is unknown,
109// its tag value is changed to a single question mark.
110template <bool unknown_to_question_mark = false, typename TCont>
111void encode_tag_pair(std::string_view tag, std::string_view value,
112 TCont& dest) {
113 dest.push_back('[');
114 dest.insert(dest.end(), tag.begin(), tag.end());
115 dest.push_back('\0');
116
117 dest.push_back('"');
118 if (unknown_to_question_mark && value.empty() &&
119 (tag == "Event" || tag == "Site" || tag == "Round" || tag == "White" ||
120 tag == "Black")) {
121 dest.push_back('?');
122 } else {
123 auto value_begin = dest.size();
124 dest.insert(dest.end(), value.begin(), value.end());
125 escape_string(dest, value_begin);
126 }
127 dest.push_back('"');
128
129 dest.push_back(']');
130 // "Each tag pair should appear left justified on a line by itself"
131 dest.push_back('\n');
132}

134// Encode a comment as "rest of the line": this comment type starts with a
135// semicolon character and continues to the end of the line.
136// If @e comment include any newline or line break char the comment cannot be
137// encoded in this way: it return false and does not modify @e dest.
138template <int hard_len = 0, typename TCont>
139[[nodiscard]] bool encode_comment_rest_of_line(std::string_view comment,
140 TCont& dest) {
141 if ((hard_len != 0 && comment.size() >= hard_len) ||
142 std::any_of(comment.begin(), comment.end(),
143 [](char ch) { return ch == '\n' || ch == '\0'; }))
144 return false;
145
146 if (!dest.empty() && dest.back() != '\0' && dest.back() != '\n') {
147 dest.push_back('\0');
148 }
149 dest.push_back(';');
150 dest.insert(dest.end(), comment.begin(), comment.end());
151 dest.push_back('\n');
152 return true;
153}

155// Encode a comment in one of the two kinds specified by PGN standard.
156// The kind that "starts with a left brace character and continues to the next
157// right brace character" is preferred and used if the comment do not contains
158// curly braces itself. If the comments contains both curly braces and newline
159// or line break chars the curly braces inside the comment are replaced with
160// UTF-8 fullwidth curly braces.
161template <int hard_len = 0, typename TCont>
162static void encode_comment(std::string_view comment, TCont& dest) {
163 auto is_curly = [](char ch) { return ch == '{' || ch == '}'; };
164 auto it_curly = std::find_if(comment.begin(), comment.end(), is_curly);
165 if (it_curly != comment.end() &&
166 encode_comment_rest_of_line<hard_len>(comment, dest))
167 return;
168
169 dest.push_back('{');
170 dest.insert(dest.end(), comment.begin(), comment.end());
171 if (it_curly != comment.end()) {
172 // Replace curly braces with UTF-8 fullwidth curly braces U+FF5B
173 // (ef bd 9b) or U+FF5D (ef bd 9d).
174 auto it = dest.end() - std::distance(it_curly, comment.end());
175 do {
176 auto replace_char = (*it == '{') ? u8"\uFF5B" : u8"\uFF5D";
177 static_assert(std::u8string_view(u8"\uFF5D").size() == 3);
178 it = dest.insert(it, 2, '\0');
179 it = std::copy_n(replace_char, 3, it);
180
181 it = std::find_if(it, dest.end(), is_curly);
182 } while (it != dest.end());
183 }
184 dest.push_back('}');
185 dest.push_back('\0');
186}

188// Encode the movetext section according to the PGN standard.
189// @param m: iterator to the list of moves.
190// @param initial_ply: the ply of the initial position.
191// Should be even if it is white turn to move.
192// @param dest: the container where movetext section will be appended.
193template <int hard_len = 0, typename Iter, typename TCont>
194void encode_movetext(Iter m, long long initial_ply, TCont& dest) {
195 std::vector<long long> ply = {initial_ply};
196 auto move_end = dest.size();
197 dest.push_back('\n');
198
199 // Check if there is a pre-game comment
200 if (!m->comment.empty())
201 encode_comment<hard_len>(m->comment, dest);
202
203 while ((m = m->nextMoveInPGN())) {
204 if (m->startMarker()) {
205 ply.push_back(ply.back() - 1);
206 dest.push_back('(');
207 if (!m->comment.empty())
208 encode_comment<hard_len>(m->comment, dest);
209
210 } else if (m->endMarker()) {
211 if (m->nextMoveInPGN()) {
212 ply.pop_back();
213 if (dest.back() == '\0') {
214 dest.back() = ')';
215 } else {
216 dest.push_back(')');
217 }
218 dest.push_back('\0');
219 }
220
221 } else {
222 auto white_to_move = (ply.back() % 2) == 0;
223 if (white_to_move || move_end != dest.size()) {
224 auto move_number = std::to_string(ply.back() / 2 + 1);
225 move_number.append(white_to_move ? 1 : 3, '.');
226 dest.insert(dest.end(), move_number.begin(), move_number.end());
227 }
228 std::string_view san = m->san;
229 dest.insert(dest.end(), san.begin(), san.end());
230 dest.push_back('\0');
231 move_end = dest.size();
232 ply.back()++;
233
234 for (int i = 0, n = m->nagCount; i < n; ++i) {
235 dest.push_back('$');
236 auto nag_str = std::to_string(m->nags[i]);
237 dest.insert(dest.end(), nag_str.begin(), nag_str.end());
238 dest.push_back('\0');
239 }
240 if (!m->comment.empty())
241 encode_comment<hard_len>(m->comment, dest);
242 }
243 }
244
245 if (dest.back() == '\0')
246 dest.back() = '\n';
247}

249// Encode a game according to the PGN standard.
250// @param game: the game to be encoded.
251// @param dest: the container where the PGN Game will be appended.
252template <int hard_len = 0, typename TGame, typename TCont>
253void encode_game(TGame const& game, TCont& dest) {
254 game.viewTagPairs(
255 [&](auto tag, auto value) { encode_tag_pair(tag, value, dest); });
256
257 encode_movetext<hard_len>(game.movetree(), game.initialPlyCounter(), dest);
258
259 auto result = game.GetResultStr();
260 dest.insert(dest.end(), result.begin(), result.end());
261 dest.push_back('\n');
262}

264// Encode a game according to the PGN standard, adding newline chars to make it
265// more readable.
266// @param game: the game to be encoded.
267// @param dest: the container where the PGN Game will be appended.
268template <int desired_len = 80, typename TGame, typename TCont>
269void encode(TGame const& game, TCont& dest) {
270 auto begin = dest.size();
271 encode_game(game, dest);
272 break_lines<desired_len>(dest.begin() + begin, dest.end());
273}
