88inline bool is_PGNsymbol(unsigned char ch) {
89 constexpr unsigned long long tok_map[] = {init_symbol_map(0),
90 init_symbol_map(1)};
91 auto high = ch / 64;
92 auto low = ch % 64;
93 return high > 1 ? false : tok_map[high] & (1ULL << low);
94}

127template <typename TView> char is_PGNtermination(TView tok) {
128 auto n_chars = std::distance(tok.first, tok.second);
129 if (n_chars == 3) {
130 if (std::equal(tok.first, tok.first + 3, "1-0"))
131 return '1';
132 if (std::equal(tok.first, tok.first + 3, "0-1"))
133 return '0';
134 if (std::equal(tok.first, tok.first + 3, "1/2"))
135 return '/';
136 if (std::equal(tok.first, tok.first + 3, "1:0"))
137 return '1';
138 if (std::equal(tok.first, tok.first + 3, "0:1"))
139 return '0';
140 } else if (n_chars == 7) {
141 if (std::equal(tok.first, tok.first + 7, "1/2-1/2") ||
142 std::equal(tok.first, tok.first + 7, "1/2:1/2"))
143 return '/';
144 }
145 return 0;
146}

158template <typename TInput, typename TVisitor>
159bool parse_token(char ch, TInput& input, TVisitor& parser, int& section) {
160 switch (ch) {
161 case ' ': // self terminating
162 case '\t': // self terminating
163 case '\v': // self terminating
164 case '\r': // self terminating
165 return true;
166
167 case '\n': // self terminating
168 return parser.visitPGN_EndOfLine();
169
170 case '.': // self terminating
171 return true;
172
173 case '<': // self terminating
174 return true;
175
176 case '>': // self terminating
177 return true;
178
179 case '*': // self terminating
180 parser.visitPGN_ResultFinal('*');
181 return false;
182
183 case '(': // self terminating
184 return parser.visitPGN_VariationStart();
185
186 case ')': // self terminating
187 return parser.visitPGN_VariationEnd();
188
189 case '[': // --> ']', can span multiple lines
190 if (section <= 0) {
191 section = 0;
192 auto skip_spaces = [&]() {
193 auto spaces = input.read_while(is_PGNwhitespace);
194 while (spaces.first != spaces.second) {
195 if (*spaces.first++ == '\n')
196 parser.visitPGN_EndOfLine();
197 }
198 };
199 skip_spaces();
200 auto tag = input.read_while(is_PGNsymbol);
201
202 skip_spaces();
203 auto value = input.read_until(']');
204
205 // Remove the " char at the start and deal with the special case of
206 // a ] char inside the string token.
207 if (value.first != value.second && *value.first == '"') {
208 auto is_terminated = [&]() {
209 for (auto it = value.first; it != value.second; ++it) {
210 if (*it == '"')
211 return true;
212 if (*it == '\\' && ++it == value.second)
213 break;
214 }
215 return false;
216 };
217 ++value.first;
218 while (!input.last_column() && !is_terminated()) {
219 value.second = input.read_until(']').second;
220 }
221 }
222 // trim right
223 while (value.first != value.second) {
224 auto last_ch = *--value.second;
225 if (last_ch == '"') {
226 break;
227 }
228 if (!is_PGNwhitespace(last_ch)) {
229 ++value.second;
230 break;
231 }
232 if (last_ch == '\n')
233 parser.visitPGN_EndOfLine();
234 }
235 return parser.visitPGN_TagPair(tag, value);
236 }
237 input.sungetc();
238 parser.visitPGN_inputUnexpectedPGNHeader();
239 return false;
240
241 case '{': // --> '}', can span multiple lines
242 return parser.visitPGN_Comment(input.read_until('}'));
243
244 case ';': // --> '\n'
245 return parser.visitPGN_Comment(input.read_line());
246
247 case '%': // --> '\n', only if "appearing in the first column of a line"
248 if (input.first_column()) {
249 return parser.visitPGN_Escape(input.read_line());
250 }
251 return parser.visitPGN_Unknown(
252 input.read_token([](char c) { return c == '%'; }));
253
254 case '$': // terminated just prior to the first non-digit character
255 return parser.visitPGN_NAG(input.read_token(is_PGNdigit));
256
257 case '?': // Suffix annotations: "!", "?", "!!", "!?", "?!", and "??"
258 case '!': // "At most one such suffix annotation may appear per move"
259 return parser.visitPGN_Suffix(
260 input.read_token([](char c) { return c == '!' || c == '?'; }));
261 }
262
263 // "A symbol token is terminated just prior to the first non-symbol
264 // character following the symbol character sequence."
265 auto tok = input.read_token(is_PGNsymbol);
266 bool epd = (section < 0 && std::count(tok.first, tok.second, '/') == 7);
267 section = 1;
268
269 if (epd) {
270 tok.second = input.read_line().second;
271 parser.visitPGN_EPD(tok);
272 return false;
273 }
274
275 auto notdigit = std::find_if_not(tok.first, tok.second, is_PGNdigit);
276 if (notdigit == tok.first)
277 return parser.visitPGN_SANMove(tok);
278
279 if (notdigit == tok.second)
280 return parser.visitPGN_MoveNum(tok);
281
282 if (auto result = is_PGNtermination(tok)) {
283 parser.visitPGN_ResultFinal(result);
284 return false;
285 }
286
287 return parser.visitPGN_Unknown(tok);
288}

295public:
296 InputMemory(const char* begin, const char* end)

300 char sbumpc() {
301 assert(it_ != end_);
302 return *it_++;
303 };
304
306 void sungetc() {
307 assert(it_ != begin_ && it_ != end_);
308 --it_;
309 }
310
312 bool eof() const { return it_ == end_; };
313
315 std::size_t n_read() const { return std::distance(begin_, it_); }
316
319 bool first_column() const { return (n_read() < 2 || *(it_ - 2) == '\n'); };
320
323 bool last_column() const { return eof() || *it_ == '\n' || *it_ == '\r'; }

327 std::pair<const char*, const char*> read_line() {
328 auto first = it_;
329 it_ = std::find(it_, end_, '\n');
330 return {first, it_};
331 }
332
335 std::pair<const char*, const char*> read_until(char delim) {
336 auto first = it_;
337 it_ = std::find(it_, end_, delim);
338 auto second = (it_ == end_) ? it_ : it_++;
339 return {first, second};
340 }

343 template <typename Cond>
344 std::pair<const char*, const char*> read_while(Cond cond) {
345 auto first = it_;
346 it_ = std::find_if_not(it_, end_, cond);
347 return {first, it_};
348 }

352 template <typename Cond>
353 std::pair<const char*, const char*> read_token(Cond cond) {
354 assert(it_ != begin_);
355 auto first = it_ - 1;
356 it_ = std::find_if_not(it_, end_, cond);
357 return {first, it_};
358 }

374template <typename TVisitor>
375std::pair<std::size_t, bool> parse_game(pgn_impl::InputMemory input,
376 TVisitor&& parser) {
377 int section = -1;
378 do {
379 if (input.eof()) {
380 if (section >= 0)
381 parser.visitPGN_inputEOF();
382 break;
383 }
384 } while (pgn_impl::parse_token(input.sbumpc(), input, parser, section));
385
386 return {input.n_read(), section >= 0};
387}

405template <bool unescape = false, typename TString>
406std::size_t normalize(TString& str, std::size_t pos) {
407 std::size_t n_newlines = 0;
408 for (std::size_t i = pos, n = str.size(); i < n; ++i) {
409 unsigned char ch = str[i];
410 // An invalid UTF-8 sequence is considered a Latin1 char and converted.
411 if (ch > 0xBF) {
412 unsigned char nxt = (i + 1 != n) ? str[i + 1] : 0;
413 if (nxt < 0x80 || nxt > 0xBF) {
414 str[i] = static_cast<unsigned char>(ch & 0xBF);
415 str.insert(str.begin() + i, static_cast<unsigned char>(0xC3));
416 ++i;
417 ++n;
418 }
419 } else if (ch == '\n') {
420 ++n_newlines;
421 } else if (unescape && ch == '\\' && i + 1 != n) {
422 // "A quote inside a string is represented by the backslash
423 // immediately followed by a quote. A backslash inside a string is
424 // represented by two adjacent backslashes."
425 if (str[i + 1] == '\\' || str[i + 1] == '"') {
426 str.erase(i, 1);
427 --n;
428 }
429 }
430 }
431 return n_newlines;
432}

439template <typename TView> std::size_t trim(TView& str) {
440 std::size_t n_newlines = 0;
441 auto is_space = [&n_newlines](char ch) {
442 if (ch == '\n') {
443 ++n_newlines;
444 } else if (ch != ' ' && ch != '\r' && ch != '\t' && ch != '\v') {
445 return false;
446 }
447 return true;
448 };
449 str.first = std::find_if_not(str.first, str.second, is_space);
450
451 using RevIt = std::reverse_iterator<decltype(str.first)>;
452 str.second =
453 std::find_if_not(RevIt(str.second), RevIt(str.first), is_space).base();
454
455 return n_newlines;
456}
