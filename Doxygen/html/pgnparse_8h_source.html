62 bool visitPGN_Comment(TView comment) {
63 if (nErrorsAllowed_ < 0) { // Skip until the end of the game
64 linenum_ += std::count(comment.first, comment.second, '\n');
65 return true;
66 }
67
68 linenum_ += pgn::trim(comment);
69 auto& str = game.accessMoveComment();
70 auto prevSz = str.size();
71 str.append(comment.first, comment.second);
72 linenum_ += pgn::normalize(str, prevSz);
73 return true;
74 }

76 bool visitPGN_EndOfLine() {
77 ++linenum_;
78 return true;
79 }

81 void visitPGN_EPD(TView line) {
82 ASSERT(nErrorsAllowed_ >= 0);
83 std::string tmp(line.first, line.second);
84 if (game.SetStartFen(tmp.c_str()) == OK) {
85 auto opcode = std::find_if(
86 line.first, line.second, [spaces = 0](char ch) mutable {
87 return (ch == ' ') ? spaces++ == 4 : spaces == 4;
88 });
89 visitPGN_Comment(std::make_pair(opcode, line.second));
90 } else {
91 logFatalErr("Failed to parse EPD record: ", line);
92 }
93 }

95 bool visitPGN_Escape(TView) { return true; }
96
97 bool visitPGN_MoveNum(TView) { return true; }

99 bool visitPGN_NAG(TView token) {
100 if (nErrorsAllowed_ < 0) // Skip until the end of the game
101 return true;
102
103 auto nag_code = game_parseNag(token);
104 if (nag_code == 0 || game.AddNag(nag_code) != OK)
105 return logErr("Invalid annotation symbol: ", token);
106
107 return true;
108 }

110 void visitPGN_ResultFinal(char resultCh) {
111 auto result = RESULT_None;
112 switch (resultCh) {
113 case '0':
114 result = RESULT_Black;
115 break;
116 case '1':
117 result = RESULT_White;
118 break;
119 case '/':
120 result = RESULT_Draw;
121 break;
122 default:
123 ASSERT(resultCh == '*');
124 }
125
126 auto prev_result = game.GetResult();
127 if (result != prev_result) {
128 // Use the end-of-game result instead of the header tag result
129 game.SetResult(result);
130 if (prev_result != RESULT_None && nErrorsAllowed_ >= 0)
131 logErr("Final result did not match the header tag.");
132 }
133 }
134
135 bool visitPGN_SANMove(TView tok) {
136 if (nErrorsAllowed_ < 0) // Skip until the end of the game
137 return true;
138
139 simpleMoveT sm;
140 auto err = game.GetCurrentPos()->ParseMove(&sm, tok.first, tok.second);
141 if (err != OK) {
142 if (game_parseNag(tok)) // may be 'D', 'N' or a weird suffix
143 return visitPGN_NAG(tok);
144
145 return logFatalErr("Failed to parse the move: ", tok);
146 }
147 return (game.AddMove(sm) == OK)
148 ? true
149 : logFatalErr("Failed to add the move: ", tok);
150 }
151
152 bool visitPGN_Suffix(TView token) { return visitPGN_NAG(token); }
153
154 bool visitPGN_TagPair(TView tag, TView value) {
155 linenum_ += std::count(value.first, value.second, '\n');
156 if (nErrorsAllowed_ < 0) // Skip until the end of the game
157 return true;
158
159 auto tagLen = std::distance(tag.first, tag.second);
160 auto valueLen = std::distance(value.first, value.second);
161 if (tagLen == 0 || tagLen + valueLen > 240 ||
162 !parseTagPair(tag.first, tagLen, value)) // Failure
163 {
164 const auto tag_sv = std::string_view(tag.first, tagLen);
165 const auto value_sv = std::string_view(value.first, valueLen);
166 std::string tag_parsed;
167 std::string value_parsed;
168 game.viewTagPairs([&](auto game_tag, auto game_value) {
169 if (tag_sv == game_tag) {
170 tag_parsed = game_tag;
171 value_parsed = game_value;
172 }
173 });
174 if (tag_sv != tag_parsed || value_sv != value_parsed) {
175 std::string err(tag_sv);
176 err.append(" \"");
177 err.append(value_sv);
178 err.append("\" ==> ");
179 err.append(tag_parsed);
180 err.append(" \"");
181 err.append(value_parsed);
182 err.push_back('"');
183 logWarning("Error parsing the tag pair: ",
184 {err.c_str(), err.c_str() + err.size()});
185 }
186 }
187 return true;
188 }
189
190 bool visitPGN_Unknown(TView token) {
191 if (nErrorsAllowed_ < 0) // Skip until the end of the game
192 return true;
193
194 // Accept misspelled castling moves
195 std::string tmp(token.first, token.second);
196 if (tmp == "0-0" || tmp == "00") {
197 tmp = "O-O";
198 return visitPGN_SANMove({tmp.c_str(), tmp.c_str() + 3});
199 }
200 if (tmp == "0-0-0" || tmp == "000") {
201 tmp = "O-O-O";
202 return visitPGN_SANMove({tmp.c_str(), tmp.c_str() + 5});
203 }
204
205 return logErr("Unknown token: ", token);
206 }

208 bool visitPGN_VariationStart() {
209 if (nErrorsAllowed_ < 0) // Skip until the end of the game
210 return true;
211
212 if (game.AddVariation() != OK)
213 return logFatalErr("Failed to add a new variation.");
214
215 return true;
216 }
217
218 bool visitPGN_VariationEnd() {
219 if (nErrorsAllowed_ < 0) // Skip until the end of the game
220 return true;
221
222 if (game.MoveExitVariation() != OK || game.MoveForward() != OK)
223 return logFatalErr("Failed to exit from variation.");
224
225 return true;
226 }

229 bool logWarning(const char* str1, TView str2 = {nullptr, nullptr}) {
230 errors_.emplace_back(linenum_, str1);
231 if (std::distance(str2.first, str2.second) > 200) {
232 errors_.back().second.append(str2.first, 200);
233 errors_.back().second.append("...");
234 } else {
235 errors_.back().second.append(str2.first, str2.second);
236 }
237 return true;

250 bool parseTagResult(TView str) {
251 auto len = std::distance(str.first, str.second);
252 if (len > 0 && *str.first == '*') {
253 game.SetResult(RESULT_None);
254 return true;
255 }
256 if (len >= 3) {
257 if (std::equal(str.first, str.first + 3, "1-0")) {
258 game.SetResult(RESULT_White);
259 return true;
260 }
261 if (std::equal(str.first, str.first + 3, "0-1")) {
262 game.SetResult(RESULT_Black);
263 return true;
264 }
265 if (std::equal(str.first, str.first + 3, "1/2")) {
266 game.SetResult(RESULT_Draw);
267 return true;
268 }
269 }
270 return logErr("Invalid Result tag: ", str);
271 }

273 bool parseTagPair(const char* tag, size_t tagLen, TView value) {
274 switch (tagLen) {
275 case 3:
276 if (std::equal(tag, tag + 3, "ECO")) {
277 std::string tmp{value.first, value.second};
278 game.SetEco(eco_FromString(tmp.c_str()));
279 return true;
280 }
281 if (std::equal(tag, tag + 3, "FEN")) {
282 std::string tmp{value.first, value.second};
283 return game.SetStartFen(tmp.c_str()) == OK;
284 }
285 break;
286 case 4:
287 if (std::equal(tag, tag + 4, "Date")) {
288 const auto date = date_parsePGNTag(value);
289 game.SetDate(date);
290 return !date_isPartial(date);
291 }
292 break;
293 case 6:
294 if (std::equal(tag, tag + 6, "Result"))
295 return parseTagResult(value);
296 break;
297 case 7:
298 if (std::equal(tag, tag + 7, "UTCDate") &&
299 game.GetDate() == ZERO_DATE) {
300 // Add two tags: "UTCDate" and the standard "Date".
301 const auto date = date_parsePGNTag(value);
302 if (!date_isPartial(date))
303 game.SetDate(date);
304 }
305 break;
306 case 9:
307 if (std::equal(tag, tag + 9, "EventDate")) {
308 const auto date = date_parsePGNTag(value);
309 game.SetEventDate(date);
310 return !date_isPartial(date);
311 }
312 if (std::equal(tag, tag + 9, "ScidFlags")) {
313 game.SetScidFlags(value.first,
314 std::distance(value.first, value.second));
315 return true;
316 }
317 break;
318 }
319 if (tagLen >= 8) {
320 // Look for Rating Types: only the first Rating type found for
321 // each player is added as the rating. Any extra ratings are
322 // just added as normal tags.
323 if (std::equal(tag, tag + 5, "White") && game.GetWhiteElo() == 0) {
324 auto res = game.setRating(WHITE, tag + 5, tagLen - 5, value);
325 if (res >= 0)
326 return res;
327 } else if (std::equal(tag, tag + 5, "Black") &&
328 game.GetBlackElo() == 0) {
329 auto res = game.setRating(BLACK, tag + 5, tagLen - 5, value);
330 if (res >= 0)
331 return res;
332 }
333 }
334 // The c++20 {value.first, value.second} fails on AppleClang13
335 size_t valueLen = std::distance(value.first, value.second);
336 auto& str = game.addTag({tag, tagLen}, {value.first, valueLen});
337 linenum_ += pgn::normalize<true>(str, 0);
338 return true;
339 }

345struct PgnParseLog {
346 std::string log;
347 unsigned long long n_bytes = 0;
348 unsigned long long n_lines = 0;

356 bool logGame(size_t nBytes, const PgnVisitor& visitor) {
357 ++n_games;
358 for (auto& e : visitor.errors_) {
359 log += "(game " + std::to_string(n_games);
360 log += ", line " + std::to_string(n_lines + e.first) + ") ";
361 log += e.second;
362 log += "\n";
363 }
364 n_lines += visitor.linenum_;
365 n_bytes += nBytes;
366 if (visitor.nErrorsAllowed_ < 0) {
367 log += "(game " + std::to_string(n_games);
368 log += ", line " + std::to_string(n_lines) + ") ";
369 log += "End of game, ignored the part after the last error.\n";
370 return false;
371 }
372 return true;
373 }
374};

387inline bool pgnParseGame(const char* input, size_t inputLen, Game& game,
388 PgnParseLog& log) {
389 struct VisitorNoEOF : public PgnVisitor {
390 explicit VisitorNoEOF(Game& g) : PgnVisitor(g) {}
391 void visitPGN_inputEOF() {}
392 } visitor(game);
393
394 auto parse = pgn::parse_game({input, input + inputLen}, visitor);
395 if (!log.logGame(parse.first, visitor))
396 return false;
397
398 if (parse.first == inputLen && !parse.second &&
399 *game.GetMoveComment() == '\0')
400 return false;
401
402 return true;
403}
