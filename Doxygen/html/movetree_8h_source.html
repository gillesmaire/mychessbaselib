34// move can have:
35#define MAX_NAGS 8

37// This class creates a tree graph of moves representing a game.
38//
39// For each node there are two edges: *next* (to the next move) and *varChild*
40// (to an alternative line). For example the graph for the move sequence
41// "1.d4 (1.e4 e5 ( 1...c5)) (1.c4) 1...d5 2.c4" is:
42// clang-format off
43// (START_MARKER, pgn_1) --next-> (d4, pgn_2) --next-> (d5, pgn_10) --next-> (c4, pgn_11) --next-> (END_MARKER, none)
44// |
45// |-child-> (START_MARKER, pgn_3) --next-> (e4, pgn_4) --next-> (e5, pgn_5) --next-> (END_MARKER, none)
46// | |
47// | |-child-> (START_MARKER, pgn_6) --next-> (c5, pgn_7) --next-> (END_MARKER, none)
48// |
49// |-child-> (START_MARKER, pgn_8) --next-> (c4, pgn_9) --next-> (END_MARKER, none)
50// clang-format on
51// The graph illustrates also that:
52// - each variation starts and ends with two nodes containing a START_MARKER
53// and a END_MARKER instead of a move. This implies that *next* never points
54// to a START_MARKER node and *varChild* always point to a START_MARKER.
55// - multiple variations are stored as a list of children. In the example there
56// are two alternatives to "1.d4": "1.e4" is stored as direct child of "1.d4",
57// instead "1.c4" is stored as the child of the START_MARKER of "1.e4".
58// - when represented as a PGN string, the moves creates a sequence that can be
59// counted (each new variation is counted as an additional move). This is used
60// to specify a location in the game.
61//
62struct moveT {
63 moveT* prev;
64 moveT* next;
65 moveT* varChild;
66 simpleMoveT moveData; // piece moving, target square etc
67 char san[10]; // SAN representation of move
68 markerT marker; // can be NO_MARKER, START_MARKER or END_MARKER
69 byte numVariations;
70 byte nagCount;
71 byte nags[MAX_NAGS];
72 std::string comment;

94 template <typename TNew>
95 moveT* cloneLine(moveT* parent, TNew newMove) const {
96 const moveT* orig = this;
97 moveT* top = newMove();
98 *top = *orig;
99 top->prev = parent;
100 parent = top;
101 if (orig->varChild) {
102 auto root = parent->startMarker() ? parent->prev : parent;
103 parent->varChild = orig->varChild->cloneLine(root, newMove);
104 }
105
106 while (orig->next) {
107 orig = orig->next;
108 auto copy = newMove();
109 *copy = *orig;
110 parent->setNext(copy);
111 parent = copy;
112 if (orig->varChild) {
113 auto root = parent->startMarker() ? parent->prev : parent;
114 parent->varChild = orig->varChild->cloneLine(root, newMove);
115 }
116 }
117 return top;
118 }

120 void swapLine(moveT& move) {
121 ASSERT(prev && move.prev);
122 ASSERT(!startMarker() && !move.startMarker());
123 ASSERT(!endMarker() && !move.endMarker());
124 // Swap lines
125 auto swap_tmp = move.prev;
126 prev->setNext(&move);
127 swap_tmp->setNext(this);
128 // Swap children
129 std::swap(varChild, move.varChild);
130 std::swap(numVariations, move.numVariations);
131 auto updateParentLink = [](moveT& parent) {
132 for (auto tmp = parent.varChild; tmp; tmp = tmp->varChild) {
133 tmp->prev = &parent;
134 }
135 };
136 updateParentLink(*this);
137 updateParentLink(move);
138 }

164 moveT* nextMoveInPGN() const {
165 if (endMarker()) {
166 auto parent = getParent();
167 auto nextVar = parent.second->varChild;
168 return (nextVar) ? nextVar
169 : (parent.first) ? parent.first->next : nullptr;
170 }
171 return (startMarker() || !varChild) ? next : varChild;
172 }

180 void insertChild(moveT* varStart, int pos) {
181 ASSERT(!this->startMarker() && !this->endMarker());
182 ASSERT(this != varStart);
183 ASSERT(varStart->startMarker());
184 auto parent = this;
185 for (; parent->varChild && pos != 0; --pos) {
186 parent = parent->varChild;
187 }
188 varStart->varChild = varStart;
189 std::swap(parent->varChild, varStart->varChild);
190
191 varStart->prev = this;
192 ++numVariations;
193 }
194
195 void appendChild(moveT* varStart) { insertChild(varStart, -1); }
196
197 void detachChild(moveT* varStart) {
198 ASSERT(this != varStart);
199 ASSERT(varStart->startMarker());
200 auto parent = this;
201 while (parent->varChild != varStart) {
202 parent = parent->varChild;
203 }
204 std::swap(parent->varChild, varStart->varChild);
205 --numVariations;
206 };
207};
