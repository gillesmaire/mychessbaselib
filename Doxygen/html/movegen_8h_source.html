27// These functions use the following move classification:
28// - a VALID move is a move that respects the basic rules for moving the pieces,
29// for example a bishop should move diagonally. Validating this type of moves
30// do not require any info about the current position.
31// - an ATTACK move is a valid move that can capture an enemy piece at the
32// destination square. It takes into account the special rule for pawns that
33// can capture only diagonally and the board position for obstacles, for
34// example a valid diagonal bishop move is an attack move only if the
35// in-between squares are empty. To validate this type of moves it is
36// necessary to know the location of empty squares in the current position.
37// - a PSEUDO-LEGAL move is an attack move, or a non-capture pawn move, or a
38// castle move.
39// - a LEGAL move is a pseudo-legal move where the destination square is not
40// occupied by a friendly piece and that do not leave the king in check.
41// To validate this type of moves it is necessary to know the type and
42// position of every piece.
43

50inline bool valid_king(squareT sqFrom, squareT sqTo) {
51 unsigned distRank = 1 + (sqTo / NSQUARES) - (sqFrom / NSQUARES);
52 unsigned distFyle = 1 + (sqTo % NSQUARES) - (sqFrom % NSQUARES);
53 return distRank <= 2 && distFyle <= 2;
54}

56inline bool valid_knight(squareT sqFrom, squareT sqTo) {
57 int distRank = (sqTo / NSQUARES) - (sqFrom / NSQUARES);
58 int distFyle = (sqTo % NSQUARES) - (sqFrom % NSQUARES);
59 int distProduct = distRank * distFyle;
60 return (distProduct == 2 || distProduct == -2);
61}

63inline int valid_slider(squareT sqFrom, squareT sqTo, pieceT pieceType) {
64 ASSERT(pieceType == QUEEN || pieceType == ROOK || pieceType == BISHOP);
65
66 int distRank = (sqTo / NSQUARES) - (sqFrom / NSQUARES);
67 int distFyle = (sqTo % NSQUARES) - (sqFrom % NSQUARES);
68
69 // Make sure the direction is valid:
70 int sqStep;
71 bool isDiagonal = false;
72 if (distRank == 0) {
73 sqStep = 1; // horizontal
74 } else if (distFyle == 0) {
75 sqStep = NSQUARES; // vertical
76 } else if (distFyle == distRank) {
77 sqStep = NSQUARES + 1;
78 isDiagonal = true;
79 } else if (distFyle == -distRank) {
80 sqStep = NSQUARES - 1;
81 isDiagonal = true;
82 } else {
83 return 0;
84 }
85 if (pieceType == ROOK && isDiagonal)
86 return 0;
87 if (pieceType == BISHOP && !isDiagonal)
88 return 0;
89
90 return sqStep;
91}

93inline bool attack_pawn(squareT sqFrom, squareT sqTo, colorT pieceCol) {
94 int distRank = (sqTo / NSQUARES) - (sqFrom / NSQUARES);
95 int distFyle = (sqTo % NSQUARES) - (sqFrom % NSQUARES);
96 if (pieceCol == WHITE && distRank != 1)
97 return false;
98 if (pieceCol == BLACK && distRank != -1)
99 return false;
100
101 return (distFyle == 1 || distFyle == -1);
102}

104template <typename TFunc>
105bool attack_slider(squareT sqFrom, squareT sqTo, pieceT pieceType,
106 TFunc isOccupied) {
107 int sqStep = valid_slider(sqFrom, sqTo, pieceType);
108 if (sqStep == 0)
109 return false;
110
111 // Make sure all the in-between squares are empty:
112 if (sqFrom > sqTo)
113 sqStep = -sqStep;
114
115 for (int sq = sqFrom + sqStep; sq != sqTo; sq += sqStep) {
116 if (isOccupied(sq))
117 return false;
118 }
119
120 return true;
121}

136template <typename TFunc>
137bool attack(squareT sqFrom, squareT sqTo, pieceT pieceCol, pieceT pieceType,
138 TFunc isOccupied) {
139 switch (pieceType) {
140 case KING:
141 return valid_king(sqFrom, sqTo);
142 case KNIGHT:
143 return valid_knight(sqFrom, sqTo);
144 case PAWN:
145 return attack_pawn(sqFrom, sqTo, pieceCol);
146 default:
147 break;
148 }
149 return attack_slider(sqFrom, sqTo, pieceType, isOccupied);
150}

152template <typename TFunc>
153inline bool pseudo_advance_pawn(squareT sqFrom, squareT sqTo, colorT pieceCol,
154 TFunc isOccupied) {
155 if ((sqTo % NSQUARES) != (sqFrom % NSQUARES) // Different file
156 || isOccupied(sqTo)) // Pawns can only capture diagonally
157 return false;
158
159 int fromRank = sqFrom / NSQUARES;
160 int distRank = (sqTo / NSQUARES) - fromRank;
161 if (pieceCol == WHITE)
162 return distRank == 1 || (distRank == 2 && fromRank == kWPHomeRank &&
163 !isOccupied(sqFrom + NSQUARES));
164
165 return distRank == -1 || (distRank == -2 && fromRank == kBPHomeRank &&
166 !isOccupied(sqFrom - NSQUARES));
167}

170bool pseudo(squareT sqFrom, squareT sqTo, colorT pieceCol, pieceT pieceType,
171 TFunc isOccupied) {

174 pseudo_advance_pawn(sqFrom, sqTo, pieceCol, isOccupied))

195template <typename TFunc>
196inline std::pair<pieceT, squareT> opens_ray(squareT sqFrom, squareT sqTo,
197 squareT sqRay, TFunc isOccupied) {
198 ASSERT(sqRay != sqFrom);
199
200 int fyleFrom = sqFrom % NSQUARES;
201 int distFyle = (sqRay % NSQUARES) - fyleFrom;
202 int distRank = (sqRay / NSQUARES) - (sqFrom / NSQUARES);
203
204 // Make sure the direction is valid:
205 int fyleEdge;
206 int sqStep;
207 pieceT pt;
208 if (distFyle == 0) {
209 sqStep = NSQUARES; // vertical
210 fyleEdge = -1;
211 pt = ROOK;
212 } else {
213 if (fyleFrom == 0 || fyleFrom == (NSQUARES - 1))
214 return {INVALID_PIECE, 0};
215
216 if (distRank == 0) {
217 sqStep = 1; // horizontal
218 fyleEdge = 0;
219 pt = ROOK;
220 } else if (distFyle == distRank) {
221 sqStep = NSQUARES + 1;
222 fyleEdge = 0;
223 pt = BISHOP;
224 } else if (distFyle == -distRank) {
225 sqStep = NSQUARES - 1;
226 fyleEdge = NSQUARES - 1;
227 pt = BISHOP;
228 } else {
229 return {INVALID_PIECE, 0};
230 }
231 }
232 if (sqFrom > sqRay) {
233 sqStep = -sqStep;
234 fyleEdge = NSQUARES - 1 - fyleEdge;
235 }
236
237 for (int sq = sqFrom + sqStep; sq != sqRay; sq += sqStep) {
238 if (sq == sqTo || isOccupied(sq))
239 return {INVALID_PIECE, 0};
240 }
241
242 for (int sq = sqFrom - sqStep; sq < NSQUARES * NSQUARES; sq -= sqStep) {
243 if (sq < 0 || sq == sqTo)
244 break;
245
246 if (isOccupied(sq))
247 return {pt, static_cast<squareT>(sq)};
248
249 if ((sq % NSQUARES) == fyleEdge)
250 break;
251 }
252 return {INVALID_PIECE, 0};
253}
