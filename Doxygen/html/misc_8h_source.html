43class StrRange {
44protected:
45 long min_;
46 long max_;
47
48protected:
49 StrRange()
50 : min_(0), max_(0) {}
51
52public:
53 explicit StrRange(const char* range) {
54 char* next;
55 min_ = std::strtol(range, &next, 10);
56 char* end;
57 max_ = std::strtol(next, &end, 10);
58 if (next == end) max_ = min_;
59 if (min_ > max_) std::swap(min_, max_);
60 }
61
63 bool inRange(long val) const {
64 if (val < min_ || val > max_) return false;
65 return true;
66 }

250inline filterOpT strGetFilterOp (const char * str)
251{
252 switch (*str) {
253 // AND:

255 // OR:
256 case 'O': case 'o': case '1': return FILTEROP_OR;
257 // RESET:

284inline uint32_t strStartHash(const char* str) {
285 ASSERT(str != 0);
286 const unsigned char* s = reinterpret_cast<const unsigned char*>(str);
287
288 uint32_t tmp = static_cast<unsigned char>(tolower(*s));
289 uint32_t result = tmp << 24;
290 if (*s == '\0') return result;
291 tmp = static_cast<unsigned char>(tolower(*++s));
292 result += tmp << 16;
293 if (*s == '\0') return result;
294 tmp = static_cast<unsigned char>(tolower(*++s));
295 result += tmp << 8;
296 if (*s == '\0') return result;
297 result += static_cast<unsigned char>(tolower(*++s));
298 return result;
299}

355inline int strCaseCompare(const char* str1, const char* str2) {
356 ASSERT(str1 != NULL && str2 != NULL);
357 const unsigned char* s1 = reinterpret_cast<const unsigned char*>(str1);
358 const unsigned char* s2 = reinterpret_cast<const unsigned char*>(str2);
359 int c1, c2;
360 do {
361 c1 = tolower(*s1++);
362 c2 = tolower(*s2++);
363 if (c1 == '\0')
364 break;
365 } while (c1 == c2);
366
367 return c1 - c2;
368}

427// WARNING: Avoid this function!
428// Considering that the sign of a char is implementation-defined [3.9.1], the
429// int conversion ("[4.7.3] If the destination type is signed, the value is
430// unchanged if it can be represented in the destination type") can yield
431// different results on different architectures or compilers.
432// A better alternative is the standard function strcmp() that returns the
433// difference between the values of the first pair of characters (both
434// interpreted as unsigned char) that differ in the strings being compared.
435inline int strCompare(const char* s1, const char* s2)
436{
437 ASSERT (s1 != NULL && s2 != NULL);
438 while (1) {
439 if (*s1 != *s2) {
440 return ((int) *s1) - ((int) *s2);
441 }
442 if (*s1 == 0)
443 break;
444 s1++; s2++;
445 }
446 return 0;
447}

468inline uint
469strPrefix (const char * s1, const char * s2)
470{
471 ASSERT (s1 != NULL && s2 != NULL);
472 uint count = 0;
473 while (*s1 == *s2) {
474 if (*s1 == 0) { // seen end of string, strings are identical
475 return count;
476 }
477 count++; s1++; s2++;
478 }
479 return count;

498// strIsCasePrefix():
499// Returns true if the prefix string is a case-insensitive
500// prefix of longStr.
501inline bool
502strIsCasePrefix (const char * prefix, const char * longStr)
503{
504 typedef unsigned char U;
505 while (*prefix) {
506 if (*longStr == 0) { return false; }
507 if (tolower(U(*prefix)) != tolower(U(*longStr))) { return false; }
508 prefix++;
509 longStr++;
510 }
511 return true;
512}

514//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
515// strIsAlphaPrefix():
516// Returns true if the prefix string is a prefix of longStr.
517// Unlike strIsPrexix(), this version is case-insensitive and
518// spaces are ignored.
519// Example: strIsAlphaPrefix ("smith,j", "Smith, John") == true.
520inline bool
521strIsAlphaPrefix (const char * prefix, const char * longStr)
522{
523 typedef unsigned char U;
524 while (*prefix) {
525 while (*prefix == ' ') { prefix++; }
526 while (*longStr == ' ') { longStr++; }
527 if (*longStr == 0) { return false; }
528 if (tolower(U(*prefix)) != tolower(U(*longStr))) { return false; }
529 prefix++;
530 longStr++;
531 }
532 return true;
533}

575//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
576// strTrimRight():
577// Trims the provided string in-place, removing the
578// end characters that match the trimChars.
579// Returns the number of characters trimmed.
580// E.g., strTrimRight("abcyzyz", "yz") would leave the string
581// as "abc".
582inline void strTrimRight(char* target, const char* trimChars, size_t nTrimCh) {
583 const char* endTrim = trimChars + nTrimCh;
584 size_t iCh = strlen(target);
585 for (; iCh > 0; --iCh) {
586 if (std::find(trimChars, endTrim, target[iCh - 1]) == endTrim)
587 break;
588 }
589 target[iCh] = '\0';
590}
591inline void strTrimRight(char* target) {
592 return strTrimRight(target, " \t\r\n", 4);
593}
