337 std::pair<errorT, size_t>
338 stripGames(HFilter hfilter, const Progress& progress,
339 std::vector<std::string_view> const& removeTags) {
340 if (auto errModify = beginTransaction())
341 return {errModify, 0};
342
343 std::vector<std::pair<std::string_view, std::string_view>> tagsBuf;
344 std::vector<byte> encodeBuf;
345 size_t nCorrections = 0;
346 size_t iProg = 0;
347 const size_t totProg = hfilter->size();
348 errorT err = OK;
349 for (const auto gnum : hfilter) {
350 if ((++iProg % 1024 == 0) && !progress.report(iProg, totProg)) {
351 err = ERROR_UserCancel;
352 break;
353 }
354
355 bool changed = false;
356 tagsBuf.clear();
357 IndexEntry const& ie = *getIndexEntry(gnum);
358 auto gamedata = getGame(ie);
359 auto err = gamedata.decodeTags(
360 [&](auto const& tag, auto const& value) {
361 if (std::find(removeTags.begin(), removeTags.end(), tag) !=
362 removeTags.end())
363 changed = true;
364 else
365 tagsBuf.emplace_back(tag, value);
366 });
367 if (err != OK)
368 break;
369
370 if (!changed)
371 continue;
372
373 encodeBuf.clear();
374 encodeTags(tagsBuf, encodeBuf);
375 encodeBuf.insert(encodeBuf.end(), gamedata.data(),
376 gamedata.data() + gamedata.size());
377 err = codec_->saveGame(ie, tagRoster(ie),
378 {encodeBuf.data(), encodeBuf.size()}, gnum);
379 if (err != OK)
380 break;
381
382 ++nCorrections;
383 }
384 const auto err_trans = endTransaction();
385 if (err == OK)
386 err = err_trans;
387 return {err, nCorrections};
388 }

460 template <typename TOper>
461 std::pair<errorT, size_t>
462 transformIndex_(HFilter hfilter, const Progress& progress, TOper entry_op) {
463 size_t nCorrections = 0;
464 size_t iProg = 0;
465 size_t totProg = hfilter->size();
466 for (auto& gnum : hfilter) {
467 if ((++iProg % 8192 == 0) && !progress.report(iProg, totProg))
468 return std::make_pair(ERROR_UserCancel, nCorrections);
469
470 IndexEntry newIE = *getIndexEntry(gnum);
471 if (!entry_op(newIE))
472 continue;
473
474 auto err = codec_->saveIndexEntry(newIE, gnum);
475 if (err != OK)
476 return std::make_pair(err, nCorrections);
477
478 ++nCorrections;
479 }
480 return std::make_pair(OK, nCorrections);
481 }

484template <typename TInitFunc, typename TMapFunc>
485std::pair<errorT, size_t>
486scidBaseT::transformNames(nameT nt, HFilter hfilter, const Progress& progress,
487 const std::vector<std::string>& newNames,
488 TInitFunc initFunc, TMapFunc getNewID) {
489 if (auto errModify = beginTransaction())
490 return {errModify, 0};
491
492 std::vector<idNumberT> nameIDs(newNames.size());
493 auto it = nameIDs.begin();
494 for (auto& name : newNames) {
495 auto id = codec_->addName(nt, name.c_str());
496 if (id.first != OK) {
497 endTransaction();
498 return std::make_pair(id.first, size_t(0));
499 }
500 *it++ = id.second;
501 }
502
503 initFunc(nameIDs);
504
505 auto res = transformIndex_(hfilter, progress, [&](IndexEntry& ie) {
506 const IndexEntry& ie_const = ie;
507 idNumberT oldID;
508 idNumberT oldBlackID = 0;
509 idNumberT newBlackID = 0;
510 switch (nt) {
511 case NAME_PLAYER:
512 oldID = ie_const.GetWhite();
513 oldBlackID = ie_const.GetBlack();
514 newBlackID = getNewID(oldBlackID, ie_const);
515 break;
516 case NAME_EVENT:
517 oldID = ie_const.GetEvent();
518 break;
519 case NAME_SITE:
520 oldID = ie_const.GetSite();
521 break;
522 default:
523 ASSERT(nt == NAME_ROUND);
524 oldID = ie_const.GetRound();
525 }
526 const auto newID = getNewID(oldID, ie_const);
527 if (oldID == newID && oldBlackID == newBlackID)
528 return false;
529
530 switch (nt) {
531 case NAME_PLAYER:
532 ie.SetWhite(newID);
533 ie.SetBlack(newBlackID);
534 break;
535 case NAME_EVENT:
536 ie.SetEvent(newID);
537 break;
538 case NAME_SITE:
539 ie.SetSite(newID);
540 break;
541 default:
542 ASSERT(nt == NAME_ROUND);
543 ie.SetRound(newID);
544 }
545 return true;
546 });
547
548 auto err = endTransaction();
549 res.first = (res.first == OK) ? err : res.first;
550 return res;
551}
