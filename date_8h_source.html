24// DATE STORAGE FORMAT:
25// In memory, dates are stored in a 32-bit (4-byte) uint, of which only
26// the lowest 3 bytes need be used, with the lowest 5 bits for the
27// day, the next highest 4 bits for the month and the highest bits for
28// the year. This makes date comparisons easy: a bigger date value is
29// a more recent date. If a field is unknown, its value is set to zero.
30// On disk, the date is stored in 3 bytes.
31

52// date_GetYear():
53// Get the year from a date.
54inline uint32_t
55date_GetYear (dateT date)
56{
57 return (uint32_t) (date >> YEAR_SHIFT);

60//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
61// date_GetMonth():
62// Get the month from a date.

64date_GetMonth (dateT date)
65{
66 return (uint32_t) ((date >> MONTH_SHIFT) & 15);

123//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
124// date_EncodeFromString(): convert PGN tag string to date.
125// The date string format is: "yyyy.mm.dd".
126inline dateT
127date_EncodeFromString (const char * str)
128{
129 // Do checks on str's validity as a date string:
130 ASSERT(str != NULL);
131
132 dateT date;
133 uint32_t year, month, day;
134
135 // convert year:
136 year = std::strtoul(str, NULL, 10);
137 if (year > YEAR_MAX) { year = 0; }
138 date = year << YEAR_SHIFT;
139 while (*str != 0 && *str != '.') { str++; }
140 if (*str == '.') { str++; }
141
142 // convert month:
143 month = std::strtoul(str, NULL, 10);
144 if (month > 12) { return date; }
145 date |= (month << MONTH_SHIFT);
146 while (*str != 0 && *str != '.') { str++; }
147 if (*str == '.') { str++; }
148
149 // convert day:
150 day = std::strtoul(str, NULL, 10);
151 if (day > 31) { return date; }
152 date |= (day << DAY_SHIFT);
153
154 return date;
155}

166inline dateT date_parsePGNTag(const char* str, size_t len) {
167 auto is_digit = [](auto v) { return v >= 0 && v <= 9; };
168
169 if (len < 4 || len > 10)
170 return {};
171
172 int tmp[10];
173 std::transform(str, str + len, tmp, [](unsigned char ch) {
174 return ch - static_cast<unsigned char>('0');
175 });
176 std::fill(tmp + len, tmp + 10, -1);
177
178 uint32_t year = tmp[0] * 1000 + tmp[1] * 100 + tmp[2] * 10 + tmp[3];
179 if (!std::all_of(tmp, tmp + 4, is_digit) || year > YEAR_MAX)
180 return {};
181
182 uint32_t month = 0;
183 if (!is_digit(tmp[4]) && is_digit(tmp[5])) {
184 if (!is_digit(tmp[6])) {
185 // Accept the format YYYY.M.DD or YYYY.M.D
186 std::rotate(tmp + 5, tmp + 9, tmp + 10);
187 tmp[5] = 0;
188 }
189 month = tmp[5] * 10 + tmp[6];
190 if (month > 12)
191 month = 0;
192 }
193
194 uint32_t day = 0;
195 if (!is_digit(tmp[7]) && is_digit(tmp[8])) {
196 day = is_digit(tmp[9]) ? tmp[8] * 10 + tmp[9] : tmp[8];
197 if (day > 31)
198 day = 0;
199 }
200
201 return (year << YEAR_SHIFT) | (month << MONTH_SHIFT) | (day << DAY_SHIFT);
202}

204inline dateT date_parsePGNTag(std::pair<const char*, const char*> str) {
205 return date_parsePGNTag(str.first, std::distance(str.first, str.second));
206}
