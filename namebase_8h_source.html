47class NameBase {
48 std::vector<std::unique_ptr<const char[]>> names_[NUM_NAME_TYPES];
49 struct idxCmp {
50 bool operator()(const char* str1, const char* str2) const {
51 // *** Compatibility ***
52 // Older code used a custom StrTree class with a peculiar sorting:
53 // - the first char was interpreted as an unsigned char;
54 // - the remaining part was compared with the function
55 // strComapare(),
56 // which converts the chars to ints, and is not consistent with
57 // the standard function strcmp().
58 // The old StrTree class did also have unpredictable behaviors when
59 // fed with names not sorted according to that criteria, for example
60 // it could create Namebase objects with duplicate entries.
61 // ***
62 if (*str1 == *str2)
63 return strCompare(str1, str2) < 0;
64
65 return static_cast<uint32_t>(*str1) < static_cast<uint32_t>(*str2);
66 }
67 };
68 std::map<const char*, idNumberT, idxCmp> idx_[NUM_NAME_TYPES];
69
70public:
71 // Add a name (string) to the NameBase.
72 // @param nt: @e nameT type of the name to add.
73 // @param name: the name to add.
74 // @return the ID assigned to @e name.
75 idNumberT namebase_add(
76 nameT nt, std::string_view name,
77 std::map<const char*, idNumberT, idxCmp>::iterator* hint = nullptr) {
78 ASSERT(IsValidNameType(nt));
79 ASSERT(names_[nt].size() <= std::numeric_limits<idNumberT>::max());
80
81 char* alloc = new char[name.size() + 1];
82 std::copy_n(name.data(), name.size(), alloc);
83 alloc[name.size()] = '\0';
84 idNumberT newID = static_cast<idNumberT>(names_[nt].size());
85 names_[nt].emplace_back(alloc);
86 if (hint) {
87 idx_[nt].emplace_hint(*hint, alloc, newID);
88 } else {
89 idx_[nt].emplace(alloc, newID);
90 }
91 return newID;
92 }
93
94 // Return the ID corresponding to @e name.
95 // Add the name to the NameBase if it doesn't exists.
96 // @param nt: @e nameT type of the name.
97 idNumberT namebase_find_or_add(nameT nt, const char* name) {
98 ASSERT(IsValidNameType(nt));
99
100 auto& nb = idx_[nt];
101 auto it = nb.lower_bound(name);
102 if (it != nb.end() && !nb.key_comp()(name, it->first))
103 return it->second;
104
105 return namebase_add(nt, name, &it);
106 }
107
108 // Return the number of names stored in the NameBase.
109 // @param nt: a valid @e nameT type.
110 size_t namebase_size(nameT nt) const {
111 ASSERT(IsValidNameType(nt));
112
113 return names_[nt].size();
114 }

122 bool insert(const char* name, size_t nameLen, nameT nt, idNumberT id) {
123 if (id >= names_[nt].size())
124 names_[nt].resize(id + size_t{1});
125
126 if (names_[nt][id]) // A name with the same ID already exists
127 return false;
128
129 char* buf = new char[nameLen + 1];
130 std::copy_n(name, nameLen, buf);
131 buf[nameLen] = '\0';
132 names_[nt][id].reset(buf);
133 auto it = idx_[nt].emplace_hint(idx_[nt].end(), buf, id);
134 return it->second == id; // Check that the name doesn't already exists
135 }

149 std::vector<idNumberT> getFirstMatches(nameT nt, const char* str,
150 size_t maxMatches) const {
151 ASSERT(IsValidNameType(nt) && str != NULL);
152
153 std::vector<idNumberT> res;
154 size_t len = strlen(str);
155 for (auto it = idx_[nt].lower_bound(str);
156 it != idx_[nt].end() && res.size() < maxMatches; ++it) {
157 const char* s = it->first;
158 if (strlen(s) < len || !std::equal(str, str + len, s))
159 break;
160 res.emplace_back(it->second);
161 }
162 return res;
163 }

215 std::vector<uint32_t> generateHashMap(nameT nt) const {
216 std::vector<uint32_t> res(names_[nt].size());

218 names_[nt].begin(), names_[nt].end(), res.begin(),

227 std::array<std::vector<int>, NUM_NAME_TYPES>
228 calcNameFreq(Index const& idx) const {
229 std::array<std::vector<int>, NUM_NAME_TYPES> resVec;
230 for (nameT n = NAME_PLAYER; n < NUM_NAME_TYPES; n++) {
231 resVec[n].resize(GetNumNames(n), 0);
232 }
233 for (gamenumT i = 0, n = idx.GetNumGames(); i < n; i++) {
234 const IndexEntry* ie = idx.GetEntry(i);
235 resVec[NAME_PLAYER][ie->GetWhite()] += 1;
236 resVec[NAME_PLAYER][ie->GetBlack()] += 1;
237 resVec[NAME_EVENT][ie->GetEvent()] += 1;
238 resVec[NAME_SITE][ie->GetSite()] += 1;
239 resVec[NAME_ROUND][ie->GetRound()] += 1;
240 }
241 return resVec;
242 }

248 size_t count_invalid_ids(Index const& idx) const {
249 size_t n_invalid = 0;
250 std::array<size_t, NUM_NAME_TYPES> maxID;
251 for (auto n = nameT{}; n < NUM_NAME_TYPES; n++) {
252 maxID[n] = GetNumNames(n);
253 }
254 for (gamenumT i = 0, n = idx.GetNumGames(); i < n; i++) {
255 const IndexEntry* ie = idx.GetEntry(i);
256 n_invalid += ie->GetWhite() < maxID[NAME_PLAYER] ? 0 : 1;
257 n_invalid += ie->GetBlack() < maxID[NAME_PLAYER] ? 0 : 1;
258 n_invalid += ie->GetEvent() < maxID[NAME_EVENT] ? 0 : 1;
259 n_invalid += ie->GetSite() < maxID[NAME_SITE] ? 0 : 1;
260 n_invalid += ie->GetRound() < maxID[NAME_ROUND] ? 0 : 1;
261 }
262 return n_invalid;
263 }

279 static nameT NameTypeFromString(const char* str) {
280 if (*str == '\0')
281 return NAME_INVALID;
282 if (strIsAlphaPrefix(str, "player"))
283 return NAME_PLAYER;
284 if (strIsAlphaPrefix(str, "event"))
285 return NAME_EVENT;
286 if (strIsAlphaPrefix(str, "site"))
287 return NAME_SITE;
288 if (strIsAlphaPrefix(str, "round"))
289 return NAME_ROUND;
290 if (strIsAlphaPrefix("player", str))
291 return NAME_PLAYER;
292 if (strIsAlphaPrefix("event", str))
293 return NAME_EVENT;
294 if (strIsAlphaPrefix("site", str))
295 return NAME_SITE;
296 if (strIsAlphaPrefix("round", str))
297 return NAME_ROUND;
298 return NAME_INVALID;
299 }

316 res.site = nb.GetName(NAME_SITE, ie.GetSite());
317 res.white = nb.GetName(NAME_PLAYER, ie.GetWhite());

323 template <typename TEntry, typename Fn>
324 auto map(TEntry& dest, Fn getID) const {
325 {
326 auto [err, id] = getID(NAME_EVENT, event);
327 if (err)
328 return err;
329 dest.SetEvent(id);
330 }
331 {
332 auto [err, id] = getID(NAME_SITE, site);
333 if (err)

341 dest.SetRound(id);
342 }
343 {
344 auto [err, id] = getID(NAME_PLAYER, white);
345 if (err)
346 return err;
347 dest.SetWhite(id);
348 }
349 {
350 auto [err, id] = getID(NAME_PLAYER, black);
351 if (!err)
