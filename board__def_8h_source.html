45// PIECES:
46// Note that color(x) == ((x & 0x8) >> 3) and type(x) == (x & 0x7)
47// EMPTY is deliberately nonzero, and END_OF_BOARD is zero, so that
48// a board can be used as a regular 0-terminated string, provided
49// that board[NULL_SQUARE] == END_OF_BOARD, as it always should be.
50const pieceT EMPTY = 7;
51const pieceT END_OF_BOARD = 0;
52const pieceT WK = 1, WQ = 2, WR = 3, WB = 4, WN = 5, WP = 6;
53const pieceT BK = 9, BQ = 10, BR = 11, BB = 12, BN = 13, BP = 14;
54
55inline colorT piece_Color(pieceT p) {
56 return (p == EMPTY) ? NOCOLOR : ((p & 8) >> 3);
57}
58// Slightly faster piece_Color when we are sure the piece is not empty:
59inline colorT piece_Color_NotEmpty(pieceT p) { return (p & 8) >> 3; }
60
61inline pieceT piece_Type(pieceT p) { return (p & 7); }
62
63inline pieceT piece_Make(colorT c, pieceT p) { return ((c << 3) | (p & 7)); }
64
65// PIECE_CHAR[]: array of piece characters, capitals for White pieces.
66const char PIECE_CHAR[] = "xKQRBNP.xkqrbnpxMm";
67
68inline char piece_Char(pieceT p) { return PIECE_CHAR[piece_Type(p)]; }
