28struct TreeNode {
29 unsigned long long eloWhiteSum = 0; // Sum of white Elos.
30 unsigned long long eloBlackSum = 0; // Sum of bLack Elos.
31 unsigned long long yearSum = 0; // Sum of years.
32 gamenumT freq[NUM_RESULT_TYPES] = {}; // freq[0] is the total count.
33 gamenumT eloCount = 0; // Count of games with an Elo.
34 gamenumT yearCount = 0; // Count of games with year != 0.
35 FullMove move;
36
37public:
38 explicit TreeNode(FullMove m) : move(m) {}
39
40 void add(resultT result, int eloW, int eloB, unsigned year) {
41 static_assert(RESULT_None == 0);
42 freq[0]++; // total count of games
43 if (result != RESULT_None) {
44 freq[result]++;
45 }
46 if (eloW > 0 && eloB > 0) {
47 ++eloCount;
48 eloWhiteSum += eloW;
49 eloBlackSum += eloB;
50 }
51 if (year > 0) {
52 yearSum += year;
53 ++yearCount;
54 }
55 }
56
59 int score() const {
60 auto n = freq[RESULT_White] + freq[RESULT_Draw] + freq[RESULT_Black];
61 auto res = 1000ull * freq[RESULT_White] + 500ull * freq[RESULT_Draw];
62 return n ? static_cast<int>(res / n) : 500;
63 }
64
65 double eloPerformance() const {
66 if (eloCount == 0)
67 return 0;
68
69 int score = (this->score() + 5) / 10;
70 auto eloOpp = eloBlackSum;
71 if (move.getColor() != WHITE) {
72 score = 100 - score;
73 eloOpp = eloWhiteSum;
74 }
75 return 1.0 * eloOpp / eloCount + FIDE_ratingTable[score];
76 }
77
78 double avgElo() const {
79 if (eloCount == 0)
80 return 0;
81
82 auto elo = (move.getColor() == WHITE) ? eloWhiteSum : eloBlackSum;
83 return 1.0 * elo / eloCount;
84 }
85
86 double avgYear() const { return yearCount ? 1.0 * yearSum / yearCount : 0; }
87
88 double percDraws() const {
89 return freq[0] ? 100.0 * freq[RESULT_Draw] / freq[0] : 0;
90 }
91
92 static auto cmp_ngames_desc() {
93 return
94 [](auto const& a, auto const& b) { return a.freq[0] > b.freq[0]; };
95 }

154 void CacheResize(size_t max_size) {
155 Clear();
156 cache_.reserve(max_size);
157 cacheTime_.reserve(max_size);
158 }
159
160 template <typename PosT> void cacheAdd(PosT const& pos, Filter& filter) {
161 size_t idx;
162 if (cache_.size() < Size() || cache_.empty()) {
163 idx = cache_.size();
164 cache_.emplace_back();
165 cacheTime_.emplace_back();
166 } else {
167 auto it = std::min_element(cacheTime_.begin(), cacheTime_.end());
168 idx = std::distance(cacheTime_.begin(), it);
169 }
170 auto board = pos.GetBoard();
171 std::copy(board, board + 64, cache_[idx].board_);
172 cache_[idx].toMove_ = pos.GetToMove();
173 cache_[idx].cfilter_.CompressFrom(&filter);
174 cacheTime_[idx] = cacheTimeCounter_++;
175 }
176
177 template <typename PosT>
178 bool cacheRestore(PosT const& pos, Filter& filter) {
179 auto it = std::find_if(
180 cache_.begin(), cache_.end(), [&pos](auto const& e) {
181 return e.toMove_ == pos.GetToMove() &&
182 std::equal(e.board_, e.board_ + 64, pos.GetBoard());
183 });
184 if (it == cache_.end())
185 return false;
186
187 auto idx = std::distance(cache_.begin(), it);
188 if (it->cfilter_.UncompressTo(&filter) != OK) {
189 ASSERT(false); // corrupted data: should not happen
190 return false;
191 }
192 cacheTime_[idx] = cacheTimeCounter_++;
193 return true;
194 }
