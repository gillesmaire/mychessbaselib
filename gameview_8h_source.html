33class MaterialCount {
34 int8_t n_[2][8] = {};
35
36public:
38 void incr(colorT color, pieceT piece_type) {
39 ASSERT(color == 0 || color == 1);
40 ASSERT(piece_type > 0 && piece_type < 8);
41
42 ++n_[color][0];
43 ++n_[color][piece_type];
44 }
45
47 void decr(colorT color, pieceT piece_type) {
48 ASSERT(color == 0 || color == 1);
49 ASSERT(piece_type > 0 && piece_type < 8);
50
51 --n_[color][0];
52 --n_[color][piece_type];
53 }
54
56 int8_t count(colorT color) const {
57 ASSERT(color == 0 || color == 1);
58
59 return n_[color][0];
60 }
61
63 int8_t count(colorT color, pieceT piece_type) const {
64 ASSERT(color == 0 || color == 1);
65 ASSERT(piece_type > 0 && piece_type < 8);
66
67 return n_[color][piece_type];
68 }
69
70 bool operator==(const MaterialCount& b) const {
71 const int8_t* a = n_[0];
72 const int8_t* b_ptr = b.n_[0];
73 return std::equal(a, a + 16, b_ptr);
74 }
75
76 bool operator!=(const MaterialCount& b) const { return !operator==(b); }
77};

163 void Init(const Position& pos) {
164 std::fill_n(board_, 64, EMPTY_SQ_);
165 std::fill_n(*castlingRook_, 4, 0);
166
167 for (auto color : {WHITE, BLACK}) {
168 const auto pos_count = pos.GetCount(color);
169 const auto pos_list = pos.GetList(color);
170 for (uint8_t idx = 0; idx < 16; ++idx) {
171 if (idx < pos_count) {
172 const squareT sq = pos_list[idx];
173 const pieceT piece_type = piece_Type(pos.GetPiece(sq));
174 pieces_.set(color, idx, sq, piece_type);
175 board_[sq] = idx;
176 mt_.incr(color, piece_type);
177
178 if (piece_type == ROOK &&
179 square_Rank(sq) == rank_Relative(color, RANK_1)) {
180 auto oldIdx = castlingRook_[color][0];
181 if (!oldIdx || sq < pieces_.getSquare(color, oldIdx))
182 castlingRook_[color][0] = idx;
183
184 oldIdx = castlingRook_[color][1];
185 if (!oldIdx || sq > pieces_.getSquare(color, oldIdx))
186 castlingRook_[color][1] = idx;
187 }
188 } else {
189 pieces_.set(color, idx, 0, INVALID_PIECE);
190 }
191 }
192 }
193 }

195 bool isEqual(const pieceT* board, const MaterialCount& mt_count) const {
196 if (mt_ != mt_count)
197 return false;

200 const auto sq = pieces_.getSquare(WHITE, idx);
201 const auto pt = pieces_.getPieceType(WHITE, idx);
202 if (board[sq] != piece_Make(WHITE, pt))
203 return false;
204 }

207 const auto pt = pieces_.getPieceType(BLACK, idx);
208 if (board[sq] != piece_Make(BLACK, pt))
209 return false;
210 }
211 return true;
212 }

228 template <colorT color> squareT castle(bool king_side) {
229 const squareT king_to = king_side ? square_Relative(color, G1)
230 : square_Relative(color, C1);
231 const squareT rook_to = king_side ? square_Relative(color, F1)
232 : square_Relative(color, D1);
233 const auto king_idx = pieces_.getKingIdx();
234 const squareT king_from = pieces_.getSquare(color, king_idx);
235 const auto rook_idx = castlingRook_[color][king_side ? 1 : 0];
236 const squareT rook_from = pieces_.getSquare(color, rook_idx);
237
238 if (pieces_.getPieceType(color, rook_idx) != ROOK)
239 return king_from; // No rook or captured
240
241 pieces_.changeSq(color, rook_idx, rook_to);
242 pieces_.changeSq(color, king_idx, king_to);
243 board_[rook_from] = EMPTY_SQ_;
244 board_[king_from] = EMPTY_SQ_;
245 board_[rook_to] = rook_idx;
246 board_[king_to] = king_idx;
247 return rook_from;
248 }

250 template <colorT color> pieceT move(int idx, squareT to, pieceT promo) {
251 if (promo != INVALID_PIECE) {
252 pieces_.promote(color, idx, promo);
253 mt_.incr(color, promo);
254 mt_.decr(color, PAWN);
255 }
256 const auto from = pieces_.getSquare(color, idx);
257 board_[from] = EMPTY_SQ_;
258 pieces_.changeSq(color, idx, to);
259 return remove<1 - color>(to, idx);
260 }

262 template <colorT color> pieceT remove(squareT sq, int newIdx = EMPTY_SQ_) {
263 ASSERT(static_cast<uint8_t>(newIdx) == newIdx);
264 const auto oldIdx = board_[sq];
265 board_[sq] = static_cast<uint8_t>(newIdx);
266 if (oldIdx == EMPTY_SQ_)
267 return INVALID_PIECE;
268
269 pieceT removed_pt = pieces_.getPieceType(color, oldIdx);
270 mt_.decr(color, removed_pt);
271 int lastvalid_idx = mt_.count(color);
272 if (oldIdx != lastvalid_idx) {
273 squareT moved_sq = pieces_.remove(color, oldIdx, lastvalid_idx);
274 board_[moved_sq] = oldIdx;
275
276 for (auto& cRook : castlingRook_[color]) {
277 if (cRook == lastvalid_idx)
278 cRook = oldIdx;
279 }
280 }
281 return removed_pt;
282 }

290 void fillSANInfo(FullMove& lastmove) const {
291 const auto lastFrom = lastmove.getFrom();
292 const auto lastTo = lastmove.getTo();
293 const auto lastCol = lastmove.getColor();
294 auto lastPt = lastmove.getPiece();
295
296 if (lastPt == PAWN) {
297 if (lastmove.isPromo())
298 lastPt = lastmove.getPromo();
299 } else if (mt_.count(lastCol, lastPt) > 1) {
300 int ambiguity = ambiguousMove(lastFrom, lastTo, lastCol, lastPt);
301 if (ambiguity)
302 lastmove.setAmbiguity(ambiguity != 5, ambiguity >= 5);
303 }
304
305 // Look for checks
306 ASSERT(mt_.count(WHITE) >= 1 && mt_.count(BLACK) >= 1);
307
308 auto isOccupied = [this](auto sq) { return board_[sq] != EMPTY_SQ_; };
309 const auto enemyKingSq = getKingSquare(color_Flip(lastCol));
310 bool direct_check = (lastPt != KING) &&
311 movegen::attack(lastTo, enemyKingSq, lastCol,
312 lastPt, isOccupied);
313 if (direct_check || // Look for a discovered check
314 find_attacker_slider(enemyKingSq, lastCol) >= 0) {
315 lastmove.setCheck();
316
317 // TODO: Find if it's mate:
318 // - it's not mate if the king can move to a safe square
319 // - it's mate if it's double check or the attacker cannot be
320 // captured or blocked.
321 }
322 }

324private:
325 squareT getKingSquare(colorT color) const {
326 return pieces_.getSquare(color, pieces_.getKingIdx());
327 }

329 int ambiguousMove(squareT lastFrom, squareT lastTo, colorT lastCol,
330 pieceT lastPt) const {
331 int ambiguity = 0;
332
333 const squareT kingSq = getKingSquare(lastCol);
334 const colorT enemyCol = color_Flip(lastCol);
335 for (int i = 1, n = mt_.count(lastCol); i < n; i++) {
336 if (getPiece(lastCol, i) != lastPt)
337 continue; // Skip: different type
338
339 const squareT sq = getSquare(lastCol, i);
340 if (sq == lastTo)
341 continue; // Skip: this is the analyzed piece
342
343 auto isOccupied = [sq, lastFrom, this](auto square) {
344 if (square == sq)
345 return false;
346 if (square == lastFrom)
347 return true;
348 return board_[square] != EMPTY_SQ_;
349 };
350 if (!movegen::pseudo(sq, lastTo, lastCol, lastPt, isOccupied))
351 continue; // Skip: illegal move
352
353 const auto pin = movegen::opens_ray(sq, lastTo, kingSq, isOccupied);
354 if (pin.first != INVALID_PIECE) {
355 uint8_t idx = board_[pin.second];
356 if (idx != EMPTY_SQ_ && idx < mt_.count(enemyCol) &&
357 getSquare(enemyCol, idx) == pin.second) {
358 const pieceT pt = getPiece(enemyCol, idx);
359 if (pt == QUEEN || pt == pin.first)
360 continue; // Skip: pinned piece
361 }
362 }
363
364 // Ambiguity:
365 // 1 (0001) --> need from-file (preferred) or from-rank
366 // 3 (0011) --> need from-file
367 // 5 (0101) --> need from-rank
368 // 7 (0111) --> need both from-file and from-rank
369 ambiguity |= 1;
370 if (square_Rank(lastFrom) == square_Rank(sq)) {
371 ambiguity |= 2; // 0b0010
372 } else if (square_Fyle(lastFrom) == square_Fyle(sq)) {
373 ambiguity |= 4; // 0b0100
374 }
375 }
376
377 return ambiguity;
378 }
379
380 int find_attacker_slider(squareT destSq, colorT color) const {
381 for (int idx = 0, n = mt_.count(color); idx < n; ++idx) {
382 const pieceT pt = getPiece(color, idx);
383 if (pt != QUEEN && pt != ROOK && pt != BISHOP)
384 continue;
385
386 auto isOccupied = [this](auto square) {
387 return board_[square] != EMPTY_SQ_;
388 };
389 const squareT sq = getSquare(color, idx);
390 if (movegen::attack_slider(sq, destSq, pt, isOccupied)) {
391 return idx;
392 }
393 }
394 return -1;
395 }

410 template <typename FuncT> void mainLine(FuncT fn) {
411 while (const auto move = (cToMove_ == WHITE)
412 ? DecodeNextMove<FullMove, WHITE>()
413 : DecodeNextMove<FullMove, BLACK>()) {
414 cToMove_ = 1 - cToMove_;
415 if (!fn(move))
416 return;
417 }
418 }
419
420 FullMove getMove(int ply_to_skip) {
421 for (int ply = 0; ply <= ply_to_skip; ply++, cToMove_ = 1 - cToMove_) {
422 auto move = (cToMove_ == WHITE) ? DecodeNextMove<FullMove, WHITE>()
423 : DecodeNextMove<FullMove, BLACK>();
424 if (!move)
425 break;
426
427 if (ply == ply_to_skip) {
428 board_.fillSANInfo(move);
429 cToMove_ = 1 - cToMove_;
430 return move;
431 }
432 }
433 return {};
434 }
435
436 std::string getMoveSAN(int ply_to_skip, int count) {
437 std::stringstream res;
438 const auto ply_num = (cToMove_ == WHITE) ? 2 : 3;
439 for (int ply = 0; ply < ply_to_skip + count;
440 ply++, cToMove_ = 1 - cToMove_) {
441 FullMove move;
442 if (cToMove_ == WHITE) {
443 move = DecodeNextMove<FullMove, WHITE>();
444 if (!move)
445 break;
446 if (ply < ply_to_skip)
447 continue;
448 if (ply > ply_to_skip)
449 res << " ";
450 res << (ply_num + ply) / 2 << ".";
451 } else {
452 move = DecodeNextMove<FullMove, BLACK>();
453 if (!move)
454 break;
455 if (ply < ply_to_skip)
456 continue;
457 if (ply == ply_to_skip)
458 res << (ply_num + ply) / 2 << "...";
459 else
460 res << " ";
461 }
462 board_.fillSANInfo(move);
463 res << move.getSAN();
464 }
465 return res.str();
466 }

468 template <colorT toMove>
469 int search(const byte* board, const MaterialCount& mt_count) {
470 int ply = 1;
471 auto less_material = [](const MaterialCount& a, const MaterialCount& b,
472 const colorT color, const auto move) {
473 if (!move)
474 return true;
475
476 const auto captured_pt = move.getCaptured();
477 if (captured_pt == INVALID_PIECE)
478 return false;
479
480 if (a.count(color) < b.count(color))
481 return true;
482
483 return a.count(color, PAWN) + a.count(color, captured_pt) <
484 b.count(color, PAWN) + b.count(color, captured_pt);
485 };
486
487 if (cToMove_ != toMove) {
488 const auto move = DecodeNextMove<FullMove, 1 - toMove>();
489 if (!move)
490 return 0;
491 ply += 1;
492 }
493 for (;;) {
494 if (board_.isEqual(board, mt_count))
495 return ply;
496
497 {
498 const auto move = DecodeNextMove<FullMove, toMove>();
499 if (less_material(board_.materialCount(), mt_count, 1 - toMove,
500 move))
501 return 0;
502 }
503 {
504 const auto move = DecodeNextMove<FullMove, 1 - toMove>();
505 if (less_material(board_.materialCount(), mt_count, toMove,
506 move))
507 return 0;
508 }
509
510 ply += 2;
511 }
512 return 0;
513 }

515private:
516 template <typename TResult, colorT toMove> TResult DecodeNextMove() {
517 auto [err, val] = bbuf_.nextLineMove();
518 if (err)
519 return {};
520
521 return doPly<TResult, toMove>(val);
522 }
523
524 template <typename TResult, colorT toMove> TResult doPly(byte v) {
525 const auto idx_piece_moving = v >> 4;
526 const auto moving_piece = board_.getPiece(toMove, idx_piece_moving);
527 const auto from = board_.getSquare(toMove, idx_piece_moving);
528
529 auto [destSq, promo] = bbuf_.decodeMove(toMove, moving_piece, from, v);
530 if (destSq < 0 || destSq > 63)
531 return {}; // decode error
532
533 const auto to = static_cast<squareT>(destSq);
534 if (to == from) {
535 if (promo == INVALID_PIECE)
536 return {}; // decode error
537
538 if (promo == PAWN) // NULL MOVE
539 return TResult(toMove, 0, 0, KING);
540
541 // CASTLE
542 const auto rook_from = board_.castle<toMove>(promo == KING);
543 return (rook_from == from) ? TResult() // decode error
544 : TResult(toMove, from, rook_from);
545 }
546
547 bool enPassant = moving_piece == PAWN &&
548 square_Fyle(from) != square_Fyle(to);
549 pieceT captured = board_.move<toMove>(idx_piece_moving, to, promo);
550 TResult res(toMove, from, to, moving_piece);
551 if (promo != INVALID_PIECE)
552 res.setPromo(promo);
553 if (captured != INVALID_PIECE) {
554 res.setCapture(captured, false);
555 } else if (enPassant) {
556 squareT sq = (toMove == WHITE) ? to - 8 : to + 8;
557 captured = board_.remove<1 - toMove>(0x3F & sq);
558 res.setCapture(captured, true);
559 }
560 return res;
561 }
